73 int AL_add_anomaly(uint32_t group, uint32_t local)
74 {
75  const AL_AnomalyRecord* prev_anomaly_record = AL_get_latest_record();
76  AL_AnomalyRecord new_anomaly_record;
77  int ret;
78 
79  ret = AL_is_logging_enable_(group);
80  if (ret == -1) return AL_ADD_ERR_INVALID;
81  if (ret == 0) return AL_ADD_DISABLE_LOGGING;
82 
83  new_anomaly_record.time = TMGR_get_master_clock();
84  new_anomaly_record.code.group = group;
85  new_anomaly_record.code.local = local;
86  new_anomaly_record.run_length = 1;
87 
88  // 登録数上限の場合はその事実を記録する
89  if (anomaly_logger_.header == AL_RECORD_MAX - 1)
90  {
91  new_anomaly_record.code.group = AL_CORE_GROUP_ANOMALY_LOGGER;
92  new_anomaly_record.code.local = AL_FULL;
93  }
94 
95  // 同じAL_AnomalyCodeの登録は圧縮する
96  if (AC_is_equal_(&(new_anomaly_record.code), &(prev_anomaly_record->code)))
97  {
98  // 時刻と連長を更新し再登録
99  new_anomaly_record.run_length += prev_anomaly_record->run_length;
100  anomaly_logger_.records[anomaly_logger_.header - 1] = new_anomaly_record;
101  }
102  else
103  {
104  // 新規登録
105  anomaly_logger_.records[anomaly_logger_.header] = new_anomaly_record;
106 
107  // 登録数が上限でないなら先頭位置をインクリメント
108  if (anomaly_logger_.header != AL_RECORD_MAX - 1)
109  {
110  ++anomaly_logger_.header;
111  }
112  }
113 
114  ++anomaly_logger_.counter;
115 
116  if (anomaly_logger_.header == anomaly_logger_.threshold_of_nearly_full)
117  {
118  AL_add_anomaly(AL_CORE_GROUP_ANOMALY_LOGGER, AL_NEARLY_FULL);
119  }
120 
121  return AL_ADD_SUCCESS;
122 }

249 static int AL_is_logging_enable_(uint32_t group)
250 {
251  uint32_t group_idx = group / 8;
252  uint32_t group_subidx = 7 - group % 8; // 反転
253  uint8_t info;
254  uint8_t mask;
255  uint8_t ret;
256 
257  if ( !(0 <= group && group < AL_GROUP_MAX) )
258  {
259  return -1;
260  }
261 
262  info = anomaly_logger_.is_logging_enable[group_idx];
263  mask = (uint8_t)(0x01 << group_subidx);
264  ret = (uint8_t)(info & mask);
265 
266  if (ret == 0) return 0;
267  return 1;
