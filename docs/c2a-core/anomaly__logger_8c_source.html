70 int AL_add_anomaly(uint32_t group, uint32_t local)
71 {
72  const AL_AnomalyRecord* prev_anomaly_record = AL_get_latest_record();
73  AL_AnomalyRecord new_anomaly_record;
74  int ret;
75 
76  ret = AL_is_logging_enable_(group);
77  if (ret == -1) return AL_ADD_ERR_INVALID;
78  if (ret == 0) return AL_ADD_DISABLE_LOGGING;
79 
80  new_anomaly_record.time = TMGR_get_master_clock();
81  new_anomaly_record.code.group = group;
82  new_anomaly_record.code.local = local;
83  new_anomaly_record.run_length = 1;
84 
85  // 登録数上限の場合はその事実を記録する
86  if (anomaly_logger_.header == AL_RECORD_MAX - 1)
87  {
88  new_anomaly_record.code.group = AL_CORE_GROUP_ANOMALY_LOGGER;
89  new_anomaly_record.code.local = AL_FULL;
90  }
91 
92  // 同じAL_AnomalyCodeの登録は圧縮する
93  if (AC_is_equal_(&(new_anomaly_record.code), &(prev_anomaly_record->code)))
94  {
95  // 時刻と連長を更新し再登録
96  new_anomaly_record.run_length += prev_anomaly_record->run_length;
97  anomaly_logger_.records[anomaly_logger_.header - 1] = new_anomaly_record;
98  }
99  else
100  {
101  // 新規登録
102  anomaly_logger_.records[anomaly_logger_.header] = new_anomaly_record;
103 
104  // 登録数が上限でないなら先頭位置をインクリメント
105  if (anomaly_logger_.header != AL_RECORD_MAX - 1)
106  {
107  ++anomaly_logger_.header;
108  }
109  }
110 
111  ++anomaly_logger_.counter;
112 
113  if (anomaly_logger_.header == anomaly_logger_.threshold_of_nearly_full)
114  {
115  AL_add_anomaly(AL_CORE_GROUP_ANOMALY_LOGGER, AL_NEARLY_FULL);
116  }
117 
118  return AL_ADD_SUCCESS;
119 }

246 static int AL_is_logging_enable_(uint32_t group)
247 {
248  uint32_t group_idx = group / 8;
249  uint32_t group_subidx = 7 - group % 8; // 反転
250  uint8_t info;
251  uint8_t mask;
252  uint8_t ret;
253 
254  if ( !(0 <= group && group < AL_GROUP_MAX) )
255  {
256  return -1;
257  }
258 
259 
260  info = anomaly_logger_.is_logging_enable[group_idx];
261  mask = (uint8_t)(0x01 << group_subidx);
262  ret = (uint8_t)(info & mask);
263 
264  if (ret == 0) return 0;
265  return 1;
266 }
