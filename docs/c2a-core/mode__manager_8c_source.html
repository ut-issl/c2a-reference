208 static MM_ACK MM_start_transition_(MD_MODEID id)
209 {
210  size_t bc_index = MM_NOT_DEFINED;
211 
212  if (id >= MD_MODEID_MODE_MAX)
213  {
214  // 定義されていないモード番号が指定された場合
215  EL_record_event((EL_GROUP)EL_CORE_GROUP_MODE_MANAGER, MM_BAD_ID, EL_ERROR_LEVEL_LOW, (uint32_t)id);
216  return MM_BAD_ID;
217  }
218  else if (mode_manager_.stat != MM_STATUS_FINISHED)
219  {
220  // 別のモード遷移を実行中の場合
221  EL_record_event((EL_GROUP)EL_CORE_GROUP_MODE_MANAGER, MM_OVERWRITE, EL_ERROR_LEVEL_LOW, (uint32_t)mode_manager_.current_id);
222  return MM_OVERWRITE;
223  }
224 
225  // 実行するブロックコマンドを読み込んでみる
226  bc_index = mode_manager_.transition_table[mode_manager_.current_id][id];
227 
228  if (bc_index == MM_NOT_DEFINED)
229  {
230  // 実行したいモード遷移に対応するブロックコマンドが登録されていない場合
231  EL_record_event((EL_GROUP)EL_CORE_GROUP_MODE_MANAGER, MM_ILLEGAL_MOVE, EL_ERROR_LEVEL_LOW, (uint32_t)bc_index);
232  return MM_ILLEGAL_MOVE;
233  }
234 
235  // ここまでくればモード遷移を実行
236  TMGR_clear_master_mode_cycle();
237  MM_deploy_block_cmd_(mode_manager_.transition_table[mode_manager_.current_id][id]);
238  mode_manager_.previous_id = mode_manager_.current_id;
239  mode_manager_.current_id = id;
240  mode_manager_.stat = MM_STATUS_IN_PROGRESS;
241 
242  return MM_SUCCESS;
243 }

263 static MM_ACK MM_finish_transition_(void)
264 {
265  TDSP_ACK ack = TDSP_UNKNOWN;
266 
267  if (mode_manager_.stat != MM_STATUS_IN_PROGRESS)
268  {
269  // モード遷移が実行中でない場合
270  EL_record_event((EL_GROUP)EL_CORE_GROUP_MODE_MANAGER, MM_NOT_IN_PROGRESS, EL_ERROR_LEVEL_HIGH, (uint32_t)mode_manager_.current_id);
271  return MM_NOT_IN_PROGRESS;
272  }
273 
274  TMGR_clear_master_mode_cycle(); // これをstart/finish両方でしているのはなぜ？
275 
276  // タスクリストを、遷移後のモードに登録されたブロックコマンドへ更新
277  ack = TDSP_set_task_list_id(MM_get_tasklist_id_of_mode(mode_manager_.current_id));
278 
279  switch (ack)
280  {
281  case TDSP_SUCCESS:
282  break;
283 
284  default:
285  EL_record_event((EL_GROUP)EL_CORE_GROUP_MODE_MANAGER, MM_TL_LOAD_FAILED, EL_ERROR_LEVEL_HIGH, (uint32_t)ack);
286  break;
287  }
288 
289  mode_manager_.stat = MM_STATUS_FINISHED;
290 
291  return MM_SUCCESS;
292 }

307 uint16_t MM_update_transition_table_for_tlm(void)
308 {
309  uint8_t from, to;
310  uint16_t idx = 0;
311 
312  memset(mode_manager_.transition_table_for_tlm, 0x00, sizeof(mode_manager_.transition_table_for_tlm));
313 
314  for (from = 0; from < MD_MODEID_MODE_MAX; ++from)
315  {
316  for (to = 0; to < MD_MODEID_MODE_MAX; ++to)
317  {
318  if (mode_manager_.transition_table[from][to] == MM_NOT_DEFINED) continue;
319 
320  mode_manager_.transition_table_for_tlm[idx].from = from;
321  mode_manager_.transition_table_for_tlm[idx].to = to;
322  mode_manager_.transition_table_for_tlm[idx].bc_index = mode_manager_.transition_table[from][to];
323  ++idx;
324  }
325  }
326 
327  return idx;
328 }
