243 CCP_EXEC_STS Cmd_MEM_COPY_REGION_SEQ(const CommonCmdPacket* packet)
244 {
245  const uint8_t* param = CCP_get_param_head(packet);
246  uint32_t copy_width, wp;
247 
248  if (memory_dump_.rp == memory_dump_.end)
249  {
250  // 既に領域全体の読み出しが完了している場合。
251  // 処理は行わず正常終了する。
252  return CCP_EXEC_SUCCESS;
253  }
254 
255  // パラメータ読み出し。
256  endian_memcpy(&copy_width, param, 4);
257 
258  if ((memory_dump_.rp + copy_width) > memory_dump_.end)
259  {
260  // 指定コピー幅でコピーすると領域外に出てしまう場合。
261  // コピー結果が領域終端に一致するようコピー幅をクリップ。
262  copy_width = memory_dump_.end - memory_dump_.rp;
263  }
264 
265  // WP計算。WP = 宛先先頭+読み出し済みデータ量。
266  wp = memory_dump_.dest + (memory_dump_.rp - memory_dump_.begin);
267 
268  // 指定されたコピー幅だけ領域をコピーし、RPを更新。
269  memcpy((uint8_t*)wp, (const uint8_t*)memory_dump_.rp, copy_width);
270  memory_dump_.rp += copy_width;
271  return CCP_EXEC_SUCCESS;
272 }

310 static MEM_ACK MEM_form_packet_(TCP* packet,
311  uint8_t category)
312 {
313  uint32_t rp;
314  size_t len;
315  TCP_SEQ_FLAG seq_flag;
316  uint8_t* data;
317 
318  rp = memory_dump_.begin + (memory_dump_.adu_seq * MEM_DUMP_WIDTH);
319 
320  if (rp >= memory_dump_.end)
321  {
322  // Read Pointerが終端に達していれば何もせず終了
323  return MEM_NO_DATA;
324  }
325 
326  // 残りダンプ長を計算
327  len = memory_dump_.end - rp;
328 
329  if (len > MEM_DUMP_WIDTH)
330  {

332  len = MEM_DUMP_WIDTH;
333  }
334 
335  // Packet Sequence Flagの種別判定
336  seq_flag = MEM_judge_seq_flag_(rp, len);
337 
338  // メモリダンプパケットのヘッダを設定
339  MEM_setup_header_(&MEM_ctp_,
340  len,
341  category,
342  memory_dump_.adu_counter,
343  seq_flag,
344  memory_dump_.adu_seq,
345  memory_dump_.adu_size);
346 
347  // ダンプデータを設定
348  data = TCP_TLM_get_user_data_head(packet) + 5; // ADU分割によりヘッダ長が5Bytes増える
349  memcpy(data, (const void*)rp, len);

354 static TCP_SEQ_FLAG MEM_judge_seq_flag_(uint32_t rp,
355  size_t len)
356 {
357  if (memory_dump_.adu_size < MEM_DUMP_WIDTH)
358  {
359  // ADUが単パケットに収まる場合
360  return TCP_SEQ_SINGLE;
361  }
362  else
363  {
364  // ADUが複数パケットにまたがる場合
365  if (rp == memory_dump_.begin)
366  {
367  // Read Pointerが読み出し領域先頭に一致するなら開始パケット
368  return TCP_SEQ_FIRST;
369  }
370  else if (len < MEM_DUMP_WIDTH)
371  {
372  // ダンプ残量が1パケット以内に収まるなら終了パケット
373  return TCP_SEQ_LAST;
374  }
375  else
376  {
377  // それ以外の場合は中間パケット
378  return TCP_SEQ_CONT;
379  }
