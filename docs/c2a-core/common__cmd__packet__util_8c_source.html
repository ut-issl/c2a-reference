196 uint16_t CCP_get_raw_param_from_packet(const CommonCmdPacket* packet, void* dest, uint16_t max_copy_len)
197 {
198  uint16_t offset = 0;
199  CMD_CODE cmd_id = CCP_get_id(packet);
200  CCP_UTIL_ACK ack;
201  int32_t copy_len;
202 
203  if (!CA_has_raw_param(cmd_id)) return 0;
204 
205  ack = CCP_calc_param_offset_(cmd_id, (uint8_t)(CA_get_cmd_param_num(cmd_id) - 1), &offset);
206  if (ack != CCP_UTIL_ACK_OK) return 0;
207 
208  copy_len = CCP_get_param_len(packet) - offset;
209  if (copy_len < 0) return 0;
210  if (max_copy_len != 0 && copy_len > max_copy_len)
211  {
212  copy_len = max_copy_len;
213  }
214 
215  memcpy(dest, CCP_get_param_head(packet) + offset, (size_t)copy_len);
216  return (uint16_t)copy_len;
217 }

219 CCP_UTIL_ACK CCP_calc_param_offset_(CMD_CODE cmd_id, uint8_t n, uint16_t* offset)
220 {
221  uint8_t i;
222  if (cmd_id >= CA_MAX_CMDS) return CCP_UTIL_ACK_PARAM_ERR;
223  if (CA_get_cmd_param_num(cmd_id) <= n) return CCP_UTIL_ACK_PARAM_ERR;
224 
225  *offset = 0;
226  for (i = 0; i < n; ++i)
227  {
228  (*offset) += CA_get_cmd_param_size(cmd_id, i);
229  }
230  return CCP_UTIL_ACK_OK;
231 }
