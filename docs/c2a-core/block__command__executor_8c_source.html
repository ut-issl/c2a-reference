226 static CCP_EXEC_STS BCE_rotate_block_cmd_(bct_id_t block)
227 {
228  CCP_EXEC_STS ack;
229  BCE_Params* bc_exe_params;
230  BCT_Pos pos;
231 
232  if (block >= BCT_MAX_BLOCKS) return BCT_convert_bct_ack_to_ccp_exec_sts(BCT_INVALID_BLOCK_NO);
233 
234  bc_exe_params = BCE_get_bc_exe_params_(block);
235  if (!bc_exe_params->is_active) return CCP_EXEC_ILLEGAL_CONTEXT;
236  if (bc_exe_params->rotate.interval == 0) return CCP_EXEC_ILLEGAL_CONTEXT;
237 
238  ++bc_exe_params->rotate.counter;
239  if (bc_exe_params->rotate.counter < bc_exe_params->rotate.interval)
240  {
241  BCE_set_bc_exe_params_(block, bc_exe_params);
242  return CCP_EXEC_SUCCESS; // スキップ
243  }
244 
245  bc_exe_params->rotate.counter = 0;
246 
247  ++bc_exe_params->rotate.next_cmd;
248  bc_exe_params->rotate.next_cmd %= BCT_get_bc_length(block);
249 
250  BCE_set_bc_exe_params_(block, bc_exe_params);
251 
252  BCT_make_pos(&pos, block, bc_exe_params->rotate.next_cmd);
253  BCT_load_cmd(&pos, &BCE_packet_);
254  ack = PH_dispatch_command(&BCE_packet_);
255 
256  return ack;

296  }
297 
298  return CCP_EXEC_SUCCESS;

323 static CCP_EXEC_STS BCE_timelimit_combine_block_cmd_(bct_id_t block, step_t limit_step)
324 {
325  uint8_t cmd;
326  uint8_t length;
327  CCP_EXEC_STS ack;
328  BCE_Params* bc_exe_params;
329 
330  ObcTime start = TMGR_get_master_clock();
331  ObcTime finish;
332  step_t diff;
333 
334  if (block >= BCT_MAX_BLOCKS) return BCT_convert_bct_ack_to_ccp_exec_sts(BCT_INVALID_BLOCK_NO);
335 
336  bc_exe_params = BCE_get_bc_exe_params_(block);
337  if (!bc_exe_params->is_active) return CCP_EXEC_ILLEGAL_CONTEXT;
338 
339  ++bc_exe_params->timelimit_combine.call_num;
340  length = BCT_get_bc_length(block);
341 
342  // TODO:
343  // ↓ちょっとこれ微妙かも...
344  // 値が増えないように，割合が変わらないように適当に下げてる感じ
345  if (bc_exe_params->timelimit_combine.call_num >= 0xFFFF - 16)
346  {
347  bc_exe_params->timelimit_combine.call_num /= 8;
348  bc_exe_params->timelimit_combine.timeover_num /= 8;
349  }
350 
351  for (cmd = 0; cmd < length; ++cmd)
352  {
353  BCT_Pos pos;
354  pos.block = block;
355  pos.cmd = cmd;
356  BCT_load_cmd(&pos, &BCE_packet_);
357  ack = PH_dispatch_command(&BCE_packet_);
358  if (ack != CCP_EXEC_SUCCESS)
359  {
360  BCE_set_bc_exe_params_(block, bc_exe_params);
361  return ack;
362  }
363 
364  // 時間判定
365  finish = TMGR_get_master_clock();
366  diff = OBCT_diff_in_step(&start, &finish);
367  if (diff >= limit_step)
368  {
369  // 途中で中断
370  ++bc_exe_params->timelimit_combine.timeover_num;
371  bc_exe_params->timelimit_combine.last_timeover_cmd_pos = cmd;
372  if (bc_exe_params->timelimit_combine.last_timeover_cmd_pos < bc_exe_params->timelimit_combine.worst_cmd_pos)
373  {
374  bc_exe_params->timelimit_combine.worst_cmd_pos = bc_exe_params->timelimit_combine.last_timeover_cmd_pos;
375  }
376 
377  BCE_set_bc_exe_params_(block, bc_exe_params);
378  return CCP_EXEC_SUCCESS; // 異常ではないのでこれを返す
379  }
380  }
381 
382  BCE_set_bc_exe_params_(block, bc_exe_params);
383 
384  // 最後まで実行できた
385  return CCP_EXEC_SUCCESS;
386 }
