70 static void DI_GS_mst_packet_handler_(void)
71 {
72  int i;
73 
74  // 本当なら max(今の FIFO の空き, 残り時間で実行可能な数) とかしたい
75  for (i = 0; i < CCSDS_FIFO_SIZE; ++i)
76  {
77  T2M_ACK ack = T2M_form_m_pdu(&DI_GS_ms_tlm_packet_handler_.tc_packet_to_m_pdu,
78  &PH_ms_tlm_list,
79  &DI_GS_ms_tlm_packet_handler_.vcdu.m_pdu);
80  if (ack != T2M_SUCCESS) return;
81 
82  // Realtime VCDU カウンタの設定
83  VCDU_setup_realtime_vcdu_hdr(&DI_GS_ms_tlm_packet_handler_.vcdu, DI_GS_ms_tlm_packet_handler_.vcdu_counter);
84  DI_GS_ms_tlm_packet_handler_.vcdu_counter = VCDU_calc_next_counter(DI_GS_ms_tlm_packet_handler_.vcdu_counter);
85 
86  // CLCW の設定
87  // CMD の VCID と TLM の VCID は独立で関係がない
88  // TLM の VCID 種別（Realtime, Replay)によらず CLCW を設定して良い
89  // CLCW が対応する CMD の VCID は CLCW の内部で指定される
90  VCDU_set_clcw(&DI_GS_ms_tlm_packet_handler_.vcdu, GS_form_clcw());
91 
92  // 完成した VCDU を MS VCDU として送出
93  GS_send_vcdu(&gs_driver_, &DI_GS_ms_tlm_packet_handler_.vcdu);
94  }
95 }
