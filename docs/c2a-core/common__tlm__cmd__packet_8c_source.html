140 CTP_DEST_FLAG CTP_get_dest_flag(const CTCP* packet)
141 {
142  uint8_t category = TCP_TLM_get_category(packet);
143  uint8_t mask = 0xe0; // 11100000b
144 
145  category &= mask;
146  category >>= 5;
147 
148  if (category == 0)
149  {
150  return CTP_DEST_FLAG_RP;
151  }
152  else
153  {
154  return (CTP_DEST_FLAG)category;
155  }
156 }

161  uint8_t mask = 0xe0; // 11100000b
162 
163  category &= (uint8_t)(~mask);

166  TCP_TLM_set_category(packet, category);
