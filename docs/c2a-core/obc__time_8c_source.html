117 ObcTime OBCT_diff(const ObcTime* before,
118  const ObcTime* after)
119 {
120  ObcTime diff;
121 
122  // まずcycleの差分を計算
123  diff.total_cycle = after->total_cycle - before->total_cycle;
124  diff.mode_cycle = after->mode_cycle - before->mode_cycle;
125 
126  // stepのみで差分を考えればよい場合
127  if (after->step >= before->step)
128  {
129  diff.step = after->step - before->step;
130  }
131  // cycleからの桁借りが必要な場合
132  else
133  {
134  diff.step = OBCT_STEPS_PER_CYCLE - before->step + after->step;
135  --diff.total_cycle;
136  --diff.mode_cycle;
137  }
138 
139  return diff;
140 }
