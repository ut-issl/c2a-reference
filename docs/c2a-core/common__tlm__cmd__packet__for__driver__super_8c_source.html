93 DS_ERR_CODE CTCP_set_tx_frame_to_dssc(DS_StreamConfig* p_stream_config,
94  const CommonTlmCmdPacket* send_packet)
95 {
96  size_t pos;
97  size_t size;
98  uint16_t crc;
99  uint16_t packet_len = CTCP_get_packet_len(send_packet);
100  uint16_t frame_len = (uint16_t)(packet_len + EB90_FRAME_HEADER_SIZE + EB90_FRAME_FOOTER_SIZE);
101  uint8_t* tx_frame = DSSC_get_tx_frame_as_non_const_pointer(p_stream_config);
102 
103  if (frame_len > DSSC_get_tx_frame_buffer_size(p_stream_config)) return DS_ERR_CODE_ERR;
104 
105  DSSC_set_tx_frame_size(p_stream_config, frame_len);
106 
107  pos = 0;
108  size = EB90_FRAME_STX_SIZE;
109  memcpy(&(tx_frame[pos]), EB90_FRAME_kStx, size);
110  pos += size;
111  size = EB90_FRAME_LEN_SIZE;
112  ENDIAN_memcpy(&(tx_frame[pos]), &packet_len, size); // ここはエンディアンを気にする！
113  pos += size;
114 
115  size = (size_t)packet_len;
116  memcpy(&(tx_frame[pos]), send_packet->packet, size);
117  pos += size;
118 
119  crc = EB90_FRAME_calc_crc(tx_frame + EB90_FRAME_HEADER_SIZE, pos - EB90_FRAME_HEADER_SIZE);
120  size = EB90_FRAME_CRC_SIZE;
121  ENDIAN_memcpy(&(tx_frame[pos]), &crc, size); // ここはエンディアンを気にする！
122  pos += size;
123  size = EB90_FRAME_ETX_SIZE;
124  memcpy(&(tx_frame[pos]), EB90_FRAME_kEtx, size);
125 
126  return DS_ERR_CODE_OK;
127 }
