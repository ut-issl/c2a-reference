52  + A[2][0] * A[0][1] * A[1][2]
53  - A[0][0] * A[2][1] * A[1][2]
54  - A[2][0] * A[1][1] * A[0][2]
55  - A[1][0] * A[0][1] * A[2][2];
56

58  {
59  B[0][0] = (A[1][1] * A[2][2] - A[1][2] * A[2][1]);
60  B[0][1] = (A[0][2] * A[2][1] - A[0][1] * A[2][2]);
61  B[0][2] = (A[0][1] * A[1][2] - A[0][2] * A[1][1]);
62

64  B[1][1] = (A[0][0] * A[2][2] - A[0][2] * A[2][0]);

68  B[2][1] = (A[0][1] * A[2][0] - A[0][0] * A[2][1]);
