152 static void DCU_create_log_on_front_(CMD_CODE cmd_code)
153 {
154  uint8_t i;
155  uint8_t idx;
156 
157  // 最も古いログを取得
158  idx = divided_cmd_utility_.exec_log_order[DCU_LOG_MAX - 1];
159  if (divided_cmd_utility_.exec_logs[idx].status == DCU_STATUS_PROGRESS)
160  {
161  EL_record_event((EL_GROUP)EL_CORE_GROUP_DCU,
162  (uint32_t)divided_cmd_utility_.exec_logs[idx].cmd_code,
163  EL_ERROR_LEVEL_HIGH,
164  (uint32_t)divided_cmd_utility_.exec_logs[idx].exec_counter);
165  }
166 
167  for (i = (DCU_LOG_MAX - 1); i > 0; --i)
168  {
169  divided_cmd_utility_.exec_log_order[i] = divided_cmd_utility_.exec_log_order[i - 1];
170  }
171  divided_cmd_utility_.exec_log_order[0] = idx;
172 
173  DCU_clear_log_element_(idx, cmd_code);
174 }

178 {

182  DCU_move_to_front_in_log_(cmd_code);

188  DCU_clear_log_element_(idx, cmd_code);
189  }

193  }
194  else
195  {
196  // なにもしない
197  }
198 
199  divided_cmd_utility_.exec_logs[idx].last_exec_time = TMGR_get_master_clock();
200  *exec_counter = divided_cmd_utility_.exec_logs[idx].exec_counter;
201  return status;
