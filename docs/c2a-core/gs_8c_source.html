153 int GS_rec_tcf(GS_Driver* gs_driver)

159  DriverSuper* ds;
160

164  }
165  else

168  }
169

171  gs_driver->info[i].rec_status = DS_receive(ds);
172  gs_driver->info[i].ret_from_if_rx = ds->config.rec_status_.ret_from_if_rx;
173 
174  if (gs_driver->info[i].rec_status != DS_ERR_CODE_OK) continue;
175 
176  gs_driver->info[i].last_rec_time = TMGR_get_master_total_cycle();

178  for (stream = 0; stream < GS_RX_HEADER_NUM; ++stream)
179  {
180  DS_StreamConfig* p_stream_config;
181 
182  p_stream_config = &ds->stream_config[stream];
183  if (DSSC_get_rec_status(p_stream_config)->status_code != DS_STREAM_REC_STATUS_FIXED_FRAME) continue;

185  gs_driver->info[i].rec_status = DS_analyze_rec_data(ds, (uint8_t)stream, gs_driver);
186 
187  gs_driver->info[i].last_rec_tcf_type = (GS_TCF_TYPE_ENUM)stream;
188  switch (gs_driver->info[i].last_rec_tcf_type)
189  {
190  case GS_TCF_TYPE_ENUM_AD_CMD:
191  gs_driver->info[i].ad_rec_status = gs_driver->info[i].rec_status;

195  gs_driver->info[i].bc_rec_status = gs_driver->info[i].rec_status;
196  break;
197 
198  case GS_TCF_TYPE_ENUM_BD_CMD:
199  gs_driver->info[i].bd_rec_status = gs_driver->info[i].rec_status;
200  break;

203  break;
204  }
205  }
206  }

233 DS_CMD_ERR_CODE GS_send_vcdu(GS_Driver* gs_driver, const VCDU* vcdu)
234 {
235  int i;
236  DS_ERR_CODE ret_ccsds = DS_ERR_CODE_OK;
237  DS_ERR_CODE ret_uart = DS_ERR_CODE_OK;
238  size_t vcdu_size = sizeof(VCDU);
239 
240  // パディングが無ければ元を GS_tx_frame_ にコピーさせる (444Byte) のコピーが無駄
241  if (vcdu_size == VCDU_LEN)
242  {
243  DSSC_set_tx_frame(&gs_driver->driver_ccsds.super.stream_config[GS_TX_stream], (uint8_t*)vcdu);
244  DSSC_set_tx_frame(&gs_driver->driver_uart.super.stream_config[GS_TX_stream], (uint8_t*)vcdu);
245  }
246  else
247  {
248  VCDU_generate_byte_stream(vcdu, GS_tx_frame_); // 送信元にセット 消したいなぁ...
249  DSSC_set_tx_frame(&gs_driver->driver_ccsds.super.stream_config[GS_TX_stream], GS_tx_frame_);
250  DSSC_set_tx_frame(&gs_driver->driver_uart.super.stream_config[GS_TX_stream], GS_tx_frame_);
251  }
252 
253  for (i = 0; i < GS_PORT_TYPE_NUM; ++i)
254  {
255  if (i == GS_PORT_TYPE_CCSDS)
256  {
257  // CCSDS 無効, 又は バッファー空きが無いで下の処理は端折る
258  // FIXME: 一杯だった時の処理
259  gs_driver->ccsds_info.buffer_num = CCSDS_get_buffer_num();
260  if (!gs_driver->is_ccsds_tx_valid || !gs_driver->ccsds_info.buffer_num) continue;
261  }
262 
263  gs_driver->info[i].send_cycle = TMGR_get_master_total_cycle();
264  gs_driver->info[i].vcid = VCDU_get_vcid(vcdu);
265  gs_driver->info[i].vcdu_counter = VCDU_get_vcdu_counter(vcdu);
266 
267  // DS 側の名称が cmd なだけで送信しているのは TLM
268  if (i == GS_PORT_TYPE_CCSDS)
269  {
270  ret_ccsds = DS_send_general_cmd(&gs_driver->driver_ccsds.super, GS_TX_stream);
271  }
272  else
273  {
274  ret_uart = DS_send_general_cmd(&gs_driver->driver_uart.super, GS_TX_stream);
275  }
276  }
277 
278  if (ret_ccsds != DS_ERR_CODE_OK || ret_uart != DS_ERR_CODE_OK)
279  {
280  return DS_CMD_DRIVER_SUPER_ERR;
281  }
282 
283  return DS_CMD_OK;
284 }
