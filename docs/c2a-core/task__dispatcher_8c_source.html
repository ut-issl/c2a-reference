96 void TDSP_execute_pl_as_task_list(void)
97 {
98  // まずは (1)ブロックコマンドがタスクリストに展開された時のサイクル数 と (2)現在のサイクル数 を比較
99  // (1) = (2) の時は展開から 1サイクル (100ms) 以内なのでタスクリストに登録されたコマンドを一つ実行する
100  // (1) > (2) の時は今回のサイクルで実行すべきコマンドがすべて実行済みである
101  // (1) < (2) の時は展開から 1サイクル (100ms) 以上経過してしまっているので強制終了する
102 
103  if (task_dispathcer_.activated_at == TMGR_get_master_total_cycle())
104  {
105  // タスクリストの先頭コマンド実行予定時刻と現在時刻を比較
106  PL_ACK ack = PL_check_tl_cmd(&TDSP_task_list_,
107  (size_t)(TMGR_get_master_step()));
108 
109  switch (ack)
110  {
111  case PL_TLC_PAST_TIME:
112  // 実行時刻が過ぎていた場合は実行前にアノマリを登録。
113 #ifndef AL_DISALBE_AT_C2A_CORE
114  AL_add_anomaly(AL_CORE_GROUP_TASK_DISPATCHER, TDSP_STEP_OVERRUN);
115 #endif
116  EL_record_event((EL_GROUP)EL_CORE_GROUP_TASK_DISPATCHER,
117  TDSP_STEP_OVERRUN,
118  EL_ERROR_LEVEL_LOW,
119  (uint32_t)CCP_get_ti( (const CommonCmdPacket*)(PL_get_head(&TDSP_task_list_)->packet) ));
120 
121  // FALL THROUGH
122 
123  case PL_TLC_ON_TIME:
124  // 実行時刻が過ぎている、もしくは実行時刻ピッタリの場合はコマンドを実行
125  CDIS_dispatch_command(&(task_dispathcer_.tskd));
126 
127  if (task_dispathcer_.tskd.prev.cmd_ret.exec_sts != CCP_EXEC_SUCCESS)
128  {
129  // コマンド実行時に異常が発生した場合はアノマリを登録。
130 #ifndef AL_DISALBE_AT_C2A_CORE
131  AL_add_anomaly(AL_CORE_GROUP_TASK_DISPATCHER, TDSP_TASK_EXEC_FAILED);
132 #endif
133  EL_record_event((EL_GROUP)EL_CORE_GROUP_TASK_DISPATCHER,
134  TDSP_TASK_EXEC_FAILED,
135  EL_ERROR_LEVEL_HIGH,
136  task_dispathcer_.tskd.prev.cmd_ret.exec_sts);
137  }
138 
139  break;
140 
141  case PL_TLC_NOT_YET:
142  // case名が不適切だが、ここに来るのは以下の二つの場合
143  // ・タスクリストが空
144  // ・次のコマンドの実行時刻がまだ
145 
146  if (PL_count_active_nodes(&TDSP_task_list_) == 0)
147  {
148  // task_listが空なら再度タスクリストを展開
149  // これを行うとtask_dispathcer_.activated_atがインクリメントされるので、"次のサイクル実行待ち状態"になる
150  TDSP_deploy_block_as_task_list_();
151  }
152 
153  break;
154 
155  default:
156  // 基本ここには来ない
157 #ifndef AL_DISALBE_AT_C2A_CORE
158  AL_add_anomaly(AL_CORE_GROUP_TASK_DISPATCHER, TDSP_UNKNOWN);
159 #endif
160  EL_record_event((EL_GROUP)EL_CORE_GROUP_TASK_DISPATCHER,
161  TDSP_UNKNOWN,
162  EL_ERROR_LEVEL_HIGH,
163  (uint32_t)ack);
164  }
165  }
166  else if (task_dispathcer_.activated_at > TMGR_get_master_total_cycle())
167  {
168  // 次のサイクルの実行待ち状態
169  return;
170  }
171  else if (task_dispathcer_.activated_at < TMGR_get_master_total_cycle())
172  {
173  if ((task_dispathcer_.activated_at == 0) && (TMGR_get_master_total_cycle() == OBCT_MAX_CYCLE - 1))
174  {
175  // 次サイクルの実行待ち状態(サイクルオーバーフロー直前)
176  // 本来は task_dispathcer_.activated_at > TMGR_get_master_total_cycle() となるはずが、
177  // task_dispathcer_.activated_atがあふれて0に戻っている場合ここに来る
178  return;
179  }
180  else
181  {
182  // 1サイクル以内に全てのコマンドを実行し終えなかった場合ここに来る
183 #ifndef AL_DISALBE_AT_C2A_CORE
184  AL_add_anomaly(AL_CORE_GROUP_TASK_DISPATCHER, TDSP_CYCLE_OVERRUN);
185 #endif
186  EL_record_event((EL_GROUP)EL_CORE_GROUP_TASK_DISPATCHER,
187  TDSP_CYCLE_OVERRUN,
188  EL_ERROR_LEVEL_HIGH,
189  0);
190 
191  // リストをクリア->再展開し次サイクルから再実行
192  PL_clear_list(&TDSP_task_list_);
193  TDSP_deploy_block_as_task_list_();
194  }
195  }
196 }
