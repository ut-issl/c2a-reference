79 void TDSP_execute_pl_as_task_list(void)
80 {
81  // まずは (1)ブロックコマンドがタスクリストに展開された時のサイクル数 と (2)現在のサイクル数 を比較
82  // (1) = (2) の時は展開から 1サイクル (100ms) 以内なのでタスクリストに登録されたコマンドを一つ実行する
83  // (1) > (2) の時は今回のサイクルで実行すべきコマンドがすべて実行済みである
84  // (1) < (2) の時は展開から 1サイクル (100ms) 以上経過してしまっているので強制終了する
85 
86  if (TDSP_info_.activated_at == TMGR_get_master_total_cycle())
87  {
88  // タスクリストの先頭コマンド実行予定時刻と現在時刻を比較
89  PL_ACK ack = PL_check_tl_cmd(&task_list_,
90  (size_t)(TMGR_get_master_step()));
91 
92  switch (ack)
93  {
94  case PL_TLC_PAST_TIME:
95  // 実行時刻が過ぎていた場合は実行前にアノマリを登録。
96 #ifndef AL_DISALBE_AT_C2A_CORE
97  AL_add_anomaly(AL_CORE_GROUP_TASK_DISPATCHER, TDSP_STEP_OVERRUN);
98 #endif
99  EL_record_event((EL_GROUP)EL_CORE_GROUP_TASK_DISPATCHER,
100  TDSP_STEP_OVERRUN,
101  EL_ERROR_LEVEL_LOW,
102  (uint32_t)CCP_get_ti(&PL_get_head(&task_list_)->packet));
103 
104  // FALL THROUGH
105 
106  case PL_TLC_ON_TIME:
107  // 実行時刻が過ぎている、もしくは実行時刻ピッタリの場合はコマンドを実行
108  CDIS_dispatch_command(&(TDSP_info_.tskd));
109 
110  if (TDSP_info_.tskd.prev.sts != CCP_EXEC_SUCCESS)
111  {
112  // コマンド実行時に異常が発生した場合はアノマリを登録。
113 #ifndef AL_DISALBE_AT_C2A_CORE
114  AL_add_anomaly(AL_CORE_GROUP_TASK_DISPATCHER, TDSP_TASK_EXEC_FAILED);
115 #endif
116  EL_record_event((EL_GROUP)EL_CORE_GROUP_TASK_DISPATCHER,
117  TDSP_TASK_EXEC_FAILED,
118  EL_ERROR_LEVEL_HIGH,
119  (uint32_t)(TDSP_info_.tskd.prev.sts + 100)); // FIXME: CCP_EXEC_STS が負数も含むので．．．なんとかしたい
120  }
121 
122  break;
123 
124  case PL_TLC_NOT_YET:
125  // case名が不適切だが、ここに来るのは以下の二つの場合
126  // ・タスクリストが空
127  // ・次のコマンドの実行時刻がまだ
128 
129  if (PL_count_active_nodes(&task_list_) == 0)
130  {
131  // task_listが空なら再度タスクリストを展開
132  // これを行うとTDSP_info_.activated_atがインクリメントされるので、"次のサイクル実行待ち状態"になる
133  TDSP_deploy_block_as_task_list_();
134  }
135 
136  break;
137 
138  default:
139  // 基本ここには来ない
140 #ifndef AL_DISALBE_AT_C2A_CORE
141  AL_add_anomaly(AL_CORE_GROUP_TASK_DISPATCHER, TDSP_UNKNOWN);
142 #endif
143  EL_record_event((EL_GROUP)EL_CORE_GROUP_TASK_DISPATCHER,
144  TDSP_UNKNOWN,
145  EL_ERROR_LEVEL_HIGH,
146  (uint32_t)ack);
147  }
148  }
149  else if (TDSP_info_.activated_at > TMGR_get_master_total_cycle())
150  {
151  // 次のサイクルの実行待ち状態
152  return;
153  }
154  else if (TDSP_info_.activated_at < TMGR_get_master_total_cycle())
155  {
156  if ((TDSP_info_.activated_at == 0) && (TMGR_get_master_total_cycle() == OBCT_MAX_CYCLE - 1))
157  {
158  // 次サイクルの実行待ち状態(サイクルオーバーフロー直前)
159  // 本来は TDSP_info_.activated_at > TMGR_get_master_total_cycle() となるはずが、
160  // TDSP_info_.activated_atがあふれて0に戻っている場合ここに来る
161  return;
162  }
163  else
164  {
165  // 1サイクル以内に全てのコマンドを実行し終えなかった場合ここに来る
166 #ifndef AL_DISALBE_AT_C2A_CORE
167  AL_add_anomaly(AL_CORE_GROUP_TASK_DISPATCHER, TDSP_CYCLE_OVERRUN);
168 #endif
169  EL_record_event((EL_GROUP)EL_CORE_GROUP_TASK_DISPATCHER,
170  TDSP_CYCLE_OVERRUN,
171  EL_ERROR_LEVEL_HIGH,
172  0);
173 
174  // リストをクリア->再展開し次サイクルから再実行
175  PL_clear_list(&task_list_);
176  TDSP_deploy_block_as_task_list_();
177  }
178  }
