179 #endif
180  EL_record_event((EL_GROUP)EL_CORE_GROUP_APP_MANAGER, AM_NOT_REGISTERED, EL_ERROR_LEVEL_LOW, (uint32_t)id);
181  return AM_NOT_REGISTERED;

183 
184 #ifdef SILS_FW
185  app_manager_.ais[id].entry_point();
186 #else
187  start = TMGR_get_master_clock();
188  app_manager_.ais[id].entry_point();
189  finish = TMGR_get_master_clock();
190 
191  // 処理時間情報アップデート
192  app_manager_.ais[id].prev = OBCT_diff_in_step(&start, &finish);
193 
194  if (app_manager_.ais[id].max < app_manager_.ais[id].prev)
195  {
196  app_manager_.ais[id].max = app_manager_.ais[id].prev;
197  }
198 
199  if (app_manager_.ais[id].min > app_manager_.ais[id].prev)
200  {
201  app_manager_.ais[id].min = app_manager_.ais[id].prev;
202  }
203 
204 #endif
205 
206  return AM_SUCCESS;
