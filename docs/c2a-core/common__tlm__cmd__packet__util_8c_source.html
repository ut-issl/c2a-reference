199 uint16_t CCP_get_raw_param_from_packet(const CTCP* packet, void* dest, uint16_t max_copy_len)
200 {
201  uint16_t offset = 0;
202  CMD_CODE cmd_id = CCP_get_id(packet);
203  CTCP_UTIL_ACK ack;
204  int32_t copy_len;
205 
206  if (!CA_has_raw_param(cmd_id)) return 0;
207 
208  ack = CCP_calc_param_offset_(cmd_id, (uint8_t)(CA_get_cmd_param_num(cmd_id) - 1), &offset);
209  if (ack != CTCP_UTIL_ACK_OK) return 0;
210 
211  copy_len = CCP_get_param_len(packet) - offset;
212  if (copy_len < 0) return 0;
213  if (max_copy_len != 0 && copy_len > max_copy_len)
214  {
215  copy_len = max_copy_len;
216  }
217 
218  memcpy(dest, CCP_get_param_head(packet) + offset, (size_t)copy_len);
219  return (uint16_t)copy_len;
220 }

222 CTCP_UTIL_ACK CCP_calc_param_offset_(CMD_CODE cmd_id, uint8_t n, uint16_t* offset)
223 {
224  uint8_t i;
225  if (cmd_id >= CA_MAX_CMDS) return CTCP_UTIL_ACK_PARAM_ERR;
226  if (CA_get_cmd_param_num(cmd_id) <= n) return CTCP_UTIL_ACK_PARAM_ERR;
227 
228  *offset = 0;
229  for (i = 0; i < n; ++i)
230  {
231  (*offset) += CA_get_cmd_param_size(cmd_id, i);
232  }
233  return CTCP_UTIL_ACK_OK;
234 }
