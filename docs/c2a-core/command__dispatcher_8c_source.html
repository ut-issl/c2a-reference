46 void CDIS_dispatch_command(CommandDispatcher* cdis)
47 {
48  // パケットコピー用。サイズが大きいため静的変数として宣言。
49  static CommonCmdPacket packet_;
50 
51  if (cdis->lockout)
52  {
53  // 実行有効フラグが無効化されている場合は処理打ち切り。
54  return;
55  }
56 
57  if (PL_is_empty(cdis->pli))
58  {
59  // 実行すべきコマンドが無い場合は処理終了。
60  return;
61  }
62 
63  if (cdis->prev.sts != CCP_EXEC_SUCCESS)
64  {
65  // 直前コマンドが異常終了した場合は実行前に情報を保存。
66  // 実行前にコピーすることで次コマンドが異常終了した場合は
67  // prevとprev_errで2コマンド分の異常情報を保持できる。
68  cdis->prev_err = cdis->prev;
69  }
70 
71  // 実行すべきコマンドパケットを取得。
72  packet_ = PL_get_head(cdis->pli)->packet;
73 
74  // ここで実行種別を変更するのをやめた．
75  // - MOBCから配送される第二OBCにも，GS cmdやTL cmdを送信したいため
76  // - user_packet_handler での PH_user_cmd_router の dispatcher (AOBC_dispatch_command など) にて
77  // 第二OBCが受けたいコマンド種別へと変換させる．
78  /*
79  // 実行時は全てのコマンドの実行種別をRealTimeに設定する。
80  // タイムラインはここでリアルタイムに変換される。
81  // この処理は特に複数機器でパケットルーティングを行う場合重要。
82  // 普通はルーティング先はルーティング元のタイムラインを受け付けないはず。
83  CCP_set_exec_type(&packet_, CCP_EXEC_TYPE_RT);
84  */
85 
86  // 実行時情報を記録しつつコマンドを実行。
87  cdis->prev.time = TMGR_get_master_clock();
88  cdis->prev.code = CCP_get_id(&packet_);
89  cdis->prev.sts = PH_dispatch_command(&packet_);
90 
91  // 実行したコマンドをリストから破棄
92  PL_drop_executed(cdis->pli);
93 
94  if (cdis->prev.sts != CCP_EXEC_SUCCESS)
95  {
96  // 実行したコマンドが実行異常ステータスを返した場合。
97  // エラー発生カウンタをカウントアップ。
98  ++(cdis->error_counter);
99 
100  if (cdis->stop_on_error == 1)
101  {
102  // 異常時実行中断フラグが有効な場合。
103  // 実行許可フラグを無効化し以降の実行を中断。
104  cdis->lockout = 1;
105  }
106  }
107 }
