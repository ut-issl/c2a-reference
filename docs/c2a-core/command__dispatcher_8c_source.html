87 void CDIS_dispatch_command(CommandDispatcher* cdis)
88 {
89  static CommonCmdPacket packet_; // パケットコピー用．サイズが大きいため静的変数として宣言
90 
91  // 実行有効フラグが無効化されている場合は処理打ち切り
92  if (cdis->lockout) return;
93 
94  // 実行すべきコマンドが無い場合は処理終了
95  if (cdis->pl == NULL) return; // TODO: PL 側で NULL チェックに対応したら消す
96  if (PL_is_empty(cdis->pl)) return;
97 
98  if (cdis->prev.sts != CCP_EXEC_SUCCESS)
99  {
100  // 直前コマンドが異常終了した場合は実行前に情報を保存
101  // 実行前にコピーすることで次コマンドが異常終了した場合は
102  // prev と prev_err で2コマンド分の異常情報を保持できる
103  cdis->prev_err = cdis->prev;
104  }
105 
106  // 実行すべきコマンドパケットを取得
107  packet_ = *(const CommonCmdPacket*)(PL_get_head(cdis->pl)->packet);
108 
109  // ここで実行種別を変更するのをやめた．
110  // - MOBCから配送される第二OBCにも，GS cmdやTL cmdを送信したいため
111  // - user_packet_handler での PH_user_cmd_router の dispatcher (AOBC_dispatch_command など) にて
112  // 第二OBCが受けたいコマンド種別へと変換させる．
113  /*
114  // 実行時は全てのコマンドの実行種別をRealTimeに設定する。
115  // タイムラインはここでリアルタイムに変換される。
116  // この処理は特に複数機器でパケットルーティングを行う場合重要。
117  // 普通はルーティング先はルーティング元のタイムラインを受け付けないはず。
118  CCP_set_exec_type(&packet_, CCP_EXEC_TYPE_RT);
119  */
120 
121  // 実行時情報を記録しつつコマンドを実行
122  cdis->prev.time = TMGR_get_master_clock();
123  cdis->prev.code = CCP_get_id(&packet_);
124  cdis->prev.sts = PH_dispatch_command(&packet_);
125 
126  // 実行したコマンドをリストから破棄
127  PL_drop_executed(cdis->pl);
128 
129  if (cdis->prev.sts != CCP_EXEC_SUCCESS)
130  {
131  // 実行時エラー情報をELにも記録. エラー発生場所(GSCD,TLCDなど)はcdisのポインタアドレスで区別
132  uint32_t note = ((0x0000ffff & cdis->prev.code) << 16) | (0x0000ffff & cdis->prev.sts);
133  EL_record_event((EL_GROUP)EL_CORE_GROUP_CDIS_EXEC_ERR,
134  (uint32_t)cdis,
135  EL_ERROR_LEVEL_LOW,
136  note);
137 
138  // 実行したコマンドが実行異常ステータスを返した場合
139  // エラー発生カウンタをカウントアップ
140  ++(cdis->error_counter);
141 
142  if (cdis->stop_on_error == 1)
143  {
144  // 異常時実行中断フラグが有効な場合
145  // 実行許可フラグを無効化し以降の実行を中断
146  cdis->lockout = 1;
147  }
148  }
149 }
