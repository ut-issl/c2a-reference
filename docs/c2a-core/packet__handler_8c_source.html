77 static PH_ACK PH_analyze_cmd_(const CTCP* packet)
78 {
79  // ユーザー定義部
80  // 基本的には，接続されているC2Aを搭載したボードに
81  // この段階（キューイングされ，時刻調整され， PH_user_cmd_router で実行されるのではなく，この段階）で転送したいときに使う
82  PH_ACK ack = PH_user_analyze_cmd(packet);
83  if (ack != PH_UNKNOWN)
84  {
85  return ack;
86  }
87 
88  switch (CCP_get_exec_type(packet))
89  {
90  case CCP_EXEC_TYPE_GS:
91  return PH_add_gs_cmd_(packet);
92 
93  case CCP_EXEC_TYPE_TL0:
94  return PH_add_tl_cmd_(0, packet, (size_t)(TMGR_get_master_total_cycle()) );
95 
96  case CCP_EXEC_TYPE_MC:
97  // Macro Command (Block Line Command) はここで引っかかる
98  return PH_analyze_block_cmd_(packet);
99 
100  case CCP_EXEC_TYPE_RT:
101  return PH_add_rt_cmd_(packet);
102 
103  case CCP_EXEC_TYPE_TL1:
104  return PH_add_tl_cmd_(1, packet, (size_t)(TMGR_get_master_total_cycle()) );
105 
106  case CCP_EXEC_TYPE_TL2:
107  return PH_add_tl_cmd_(2, packet, (size_t)(TMGR_get_master_total_cycle()) );
108 
109  default:
110  return PH_UNKNOWN;
111  }
112 }

138 static PH_ACK PH_analyze_tlm_(const CTCP* packet)
139 {
140  CTP_DEST_FLAG flag = CTP_get_dest_flag(packet);
141 
142  // Housekeeping Telemetry
143  if (flag & CTP_DEST_FLAG_HK) PH_add_ms_tlm_(packet); // hk_tlm のフラグが立っていても，MS_TLMとして処理する方針にした
144 
145  // Mission Telemetry
146  if (flag & CTP_DEST_FLAG_MS) PH_add_ms_tlm_(packet);
147 
148  // Stored Telemetry
149  if (flag & CTP_DEST_FLAG_ST) PH_add_st_tlm_(packet);
150 
151  // Replay Telemetry
152  if (flag & CTP_DEST_FLAG_RP) PH_add_rp_tlm_(packet);
153 
154  // [TODO] 要検討:各Queue毎の登録エラー判定は未実装
155  return PH_SUCCESS;
156 }

158 CCP_EXEC_STS PH_dispatch_command(const CTCP* packet)
159 {
160  if (CTCP_get_tc_dsc(packet) != CTCP_TC_DSC_CMD)
161  {
162  // CMD以外のパケットが来たら異常判定。
163  return CCP_EXEC_PACKET_FMT_ERR;
164  }
165 
166  if (CCP_get_dest_id(packet) == CTCP_MY_DST_ID)
167  {
168  // 自分宛てのコマンドの場合は対応処理を呼び出し。
169  return CA_execute_cmd(packet);
170  }
171  else
172  {
173  // 別機器宛コマンドの場合はパケット分配処理へ
174  return PH_user_cmd_router(packet); // 当初，coreにcmd_router_が存在したが，coreにあっても意味がないのでuserのみにした
175  }
176 }
