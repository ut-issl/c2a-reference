118 PH_ACK PH_analyze_cmd_packet(const CommonCmdPacket* packet)
119 {
120  PH_ACK ack;
121  if (!CCP_is_valid_packet(packet)) return PH_ACK_INVALID_PACKET;
122 
123  // ユーザー定義部
124  // 基本的には，接続されているC2Aを搭載したボードに
125  // この段階（キューイングされ，時刻調整され， PH_user_cmd_router で実行されるのではなく，この段階）で転送したいときに使う
126  ack = PH_user_analyze_cmd(packet);
127  if (ack != PH_ACK_UNKNOWN)
128  {
129  return ack;
130  }
131 
132  switch (CCP_get_exec_type(packet))
133  {
134  case CCP_EXEC_TYPE_GS:
135  return PH_add_gs_cmd_(packet);
136 
137  case CCP_EXEC_TYPE_TL_FROM_GS:
138  return PH_add_tl_cmd_(TLCD_ID_FROM_GS, packet, TMGR_get_master_total_cycle());
139 
140  case CCP_EXEC_TYPE_BC:
141  return PH_add_block_cmd_(packet);
142 
143  case CCP_EXEC_TYPE_RT:
144  return PH_add_rt_cmd_(packet);
145 
146  case CCP_EXEC_TYPE_UTL:
147  return PH_add_utl_cmd_(TLCD_ID_FROM_GS, packet);
148 
149  case CCP_EXEC_TYPE_TL_DEPLOY_BC:
150  return PH_add_tl_cmd_(TLCD_ID_DEPLOY_BC, packet, TMGR_get_master_total_cycle());
151 
152  case CCP_EXEC_TYPE_TL_DEPLOY_TLM:
153  return PH_add_tl_cmd_(TLCD_ID_DEPLOY_TLM, packet, TMGR_get_master_total_cycle());
154 
155 #ifdef TLCD_ENABLE_MISSION_TL
156  case CCP_EXEC_TYPE_TL_FOR_MISSION:
157  return PH_add_tl_cmd_(TLCD_ID_FROM_GS_FOR_MISSION, packet, TMGR_get_master_total_cycle());
158 
159  case CCP_EXEC_TYPE_UTL_FOR_MISSION:
160  return PH_add_utl_cmd_(TLCD_ID_FROM_GS_FOR_MISSION, packet);
161 #endif
162 
163  default:
164  return PH_ACK_UNKNOWN;
165  }

194 PH_ACK PH_analyze_tlm_packet(const CommonTlmPacket* packet)
195 {
196  ctp_dest_flags_t flags;

198 
199  flags = CTP_get_dest_flags(packet);
200 
201  // FIXME: flag の match は関数化したい
202 
203  // Housekeeping Telemetry

205 
206  // Mission Telemetry
207  if (flags & CTP_DEST_FLAG_MS) PH_add_ms_tlm_(packet);
208 
209 #ifdef DR_ENABLE

213  // Replay Telemetry
214  if (flags & CTP_DEST_FLAG_RP) PH_add_rp_tlm_(packet);
215 #endif

218  return PH_ACK_SUCCESS;

286 static PH_ACK PH_add_utl_cmd_(TLCD_ID id, const CommonCmdPacket* packet)

296  CCP_copy_packet(&temp_, packet);

298  CCP_set_exec_type(&temp_, CCP_EXEC_TYPE_TL_FROM_GS); // UTL -> TL
299 
300  return PH_add_tl_cmd_(id, &temp_, TMGR_get_master_total_cycle());
