18  uint8_t category = CCP_get_param_from_packet(packet, 0, uint8_t);
19  TLM_CODE id = (TLM_CODE)CCP_get_param_from_packet(packet, 1, uint8_t);
20  uint8_t num_dumps = CCP_get_param_from_packet(packet, 2, uint8_t);
21  uint16_t len;
22  TF_TLM_FUNC_ACK ack;
23  uint8_t dr_partition_mask = 0x1f; // 00011111b // FIXME: 一時的な対応
24  uint8_t dest_flags_mask = 0xe0; // 11100000b // FIXME: 一時的な対応
25  uint8_t dr_partition;
26  ctp_dest_flags_t dest_flags;
27 
28  if (num_dumps >= 8)
29  {
30  // パケット生成回数の上限は8回とする。
31  // 32kbpsでのDL時に8VCDU/secで1秒分の通信量。
32  // これを超える場合は複数回コマンドを送信して対応する。
33  return CCP_EXEC_ILLEGAL_PARAMETER;
34  }
35 
36  // ADU生成

39  ack = TF_generate_contents(id,
40  ctp_.packet,
41  &len,
42  TSP_MAX_LEN);
43 
44  // 範囲外のTLM IDを除外
45  if (ack == TF_TLM_FUNC_ACK_NOT_DEFINED) return CCP_EXEC_ILLEGAL_PARAMETER;

47 
48  // TCPacketヘッダ設定
49  // FIXME: Space Packet 依存を直す
50  TSP_setup_primary_hdr(&ctp_, APID_MIS_TLM, len);

52  // FIXME: 他の時刻も入れる

57  dr_partition = (uint8_t)(category & dr_partition_mask);
58  dest_flags = (uint8_t)( (category & dest_flags_mask) >> 5 );
59  if (dest_flags == 0)

61  dest_flags = CTP_DEST_FLAG_RP;
62  }
63  TSP_set_dest_flgas(&ctp_, dest_flags);
64  TSP_set_dr_partition(&ctp_, dr_partition);
65  TSP_set_tlm_id(&ctp_, id);
66  TSP_set_seq_count(&ctp_, TG_get_next_adu_counter_());
67

71  PH_analyze_tlm_packet(&ctp_);
72  --num_dumps;
73  }
74
