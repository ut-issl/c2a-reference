91 static void tlc_dispatcher_(int line_no)
92 {
93  PL_ACK ack = PL_check_tl_cmd(&(PH_tl_cmd_list[line_no]),
94  (size_t)(TMGR_get_master_total_cycle()));
95 
96  switch (ack)
97  {
98  case PL_TLC_ON_TIME:
99  // 実行予定時刻ちょうど
100  CDIS_dispatch_command(&timeline_command_dispatcher_[line_no]);
101  break;
102 
103  case PL_TLC_PAST_TIME:
104 
105  // 実行予定時刻超過
106  if (timeline_command_dispatcher_[line_no].lockout == 0)
107  {
108  // Lockout無効の場合はアノマリーを記録。
109  // Lockout有効の場合はアノマリーの連続生成を防ぐため時刻超過状態でもアノマリーを記録しない。
110 #ifndef AL_DISALBE_AT_C2A_CORE
111  AL_add_anomaly(AL_CORE_GROUP_TLCD, (uint32_t)line_no);
112 #endif
113  EL_record_event((EL_GROUP)EL_CORE_GROUP_TLCD_PAST_TIME,
114  (uint32_t)line_no,
115  EL_ERROR_LEVEL_LOW,
116  (uint32_t)CCP_get_ti( (const CommonCmdPacket*)(PL_get_head(&PH_tl_cmd_list[line_no])->packet) ));
117 
118  if (timeline_command_dispatcher_[line_no].stop_on_error == 1)
119  {
120  // Lockout無効かつSOE有効の場合
121  // Lockout有効とし、タイムラインコマンド処理を停止。
122  timeline_command_dispatcher_[line_no].lockout = 1;
123  }
124  }
125 
126  // コマンド実行処理呼び出し。
127  // Lockoutとなった場合は呼び出してもコマンドは実行されない。
128  CDIS_dispatch_command(&timeline_command_dispatcher_[line_no]);
129  break;
130 
131  case PL_TLC_NOT_YET:
132  // 実行すべきコマンドなし。
133  break;
134 
135  default:
136  break;
137  }
138 }

211 static PH_ACK drop_tl_cmd_at_(int line_no, cycle_t time)
212 {
213  int i;
214 
215  PL_Node* prev = NULL;
216  PL_Node* current = (PL_Node*)PL_get_head(&(PH_tl_cmd_list[line_no])); // const_cast
217  int active_nodes_num = PL_count_active_nodes(&PH_tl_cmd_list[line_no]);
218 
219  if (current == NULL) return PH_TLC_NOT_FOUND;
220 
221  for (i = 0; i < active_nodes_num; ++i)
222  {
223  if (CCP_get_ti( (const CommonCmdPacket*)(current->packet) ) == time)
224  {
225  PL_drop_node(&(PH_tl_cmd_list[line_no]), prev, current);
226  break;
227  }
228  if (PL_get_next(current) == NULL) return PH_TLC_NOT_FOUND;
229 
230  prev = current;
231  current = current->next;
232  }
233 
234  return PH_SUCCESS;
235 }

324  return CCP_EXEC_ILLEGAL_LENGTH;
325  }
326 
327  // ライン番号を読み込み
328  line_no = param[0];
329 
330  if (line_no >= TL_ID_MAX)
331  {
332  // 指定されたライン番号が存在しない場合は異常判定
333  return CCP_EXEC_ILLEGAL_PARAMETER;
334  }
335 
336  endian_memcpy(&block_no, &param[1], SIZE_OF_BCT_ID_T);
337 
338  if (block_no >= BCT_MAX_BLOCKS)
339  {
340  // 指定されたブロック番号が存在しない場合は異常判定
341  return CCP_EXEC_ILLEGAL_PARAMETER;
342  }
343 
344  ack = PL_deploy_block_cmd(&(PH_tl_cmd_list[line_no]), block_no, (size_t)(TMGR_get_master_total_cycle()));
345 
346  if (ack != PL_SUCCESS)
347  {
348  EL_record_event((EL_GROUP)EL_CORE_GROUP_TLCD_DEPLOY_BLOCK,
349  (uint32_t)ack,
350  EL_ERROR_LEVEL_LOW,
351  (uint32_t)line_no);
352  return CCP_EXEC_ILLEGAL_CONTEXT;
353  }
354 
355  return CCP_EXEC_SUCCESS;
