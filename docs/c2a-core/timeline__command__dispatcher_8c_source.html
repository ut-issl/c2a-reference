114 static void TLCD_tlc_dispatcher_(TLCD_ID id)
115 {
116  PL_ACK ack = PL_check_tl_cmd(&(PH_tl_cmd_list[id]),
117  TMGR_get_master_total_cycle());
118 
119  switch (ack)
120  {
121  case PL_TLC_ON_TIME:
122  // 実行予定時刻ちょうど
123  CDIS_dispatch_command(&timeline_command_dispatcher_.dispatcher[id]);
124  break;
125 
126  case PL_TLC_PAST_TIME:
127 
128  // 実行予定時刻超過
129  if (timeline_command_dispatcher_.dispatcher[id].lockout == 0)
130  {
131  // Lockout無効の場合はアノマリーを記録。
132  // Lockout有効の場合はアノマリーの連続生成を防ぐため時刻超過状態でもアノマリーを記録しない。
133 #ifndef AL_DISALBE_AT_C2A_CORE
134  AL_add_anomaly(AL_CORE_GROUP_TLCD, (uint32_t)id);
135 #endif
136  EL_record_event((EL_GROUP)EL_CORE_GROUP_TLCD_PAST_TIME,
137  (uint32_t)id,
138  EL_ERROR_LEVEL_LOW,
139  (uint32_t)CCP_get_ti( (const CommonCmdPacket*)(PL_get_head(&PH_tl_cmd_list[id])->packet) ));
140 
141  if (timeline_command_dispatcher_.dispatcher[id].stop_on_error == 1)
142  {
143  // Lockout無効かつSOE有効の場合
144  // Lockout有効とし、タイムラインコマンド処理を停止。
145  timeline_command_dispatcher_.dispatcher[id].lockout = 1;
146  }
147  }
148 
149  // コマンド実行処理呼び出し。
150  // Lockoutとなった場合は呼び出してもコマンドは実行されない。
151  CDIS_dispatch_command(&timeline_command_dispatcher_.dispatcher[id]);
152  break;
153 
154  case PL_TLC_NOT_YET:
155  // 実行すべきコマンドなし。
156  break;
157 
158  default:
159  break;
160  }
161 }

225 static PH_ACK TLCD_drop_tl_cmd_at_(TLCD_ID id, cycle_t time)
226 {
227  int i;
228 
229  PL_Node* prev = NULL;
230  PL_Node* current = (PL_Node*)PL_get_head(&(PH_tl_cmd_list[id])); // const_cast
231  int active_nodes_num = PL_count_active_nodes(&PH_tl_cmd_list[id]);
232 
233  if (current == NULL) return PH_ACK_PACKET_NOT_FOUND;
234 
235  for (i = 0; i < active_nodes_num; ++i)
236  {
237  if (CCP_get_ti( (const CommonCmdPacket*)(current->packet) ) == time)
238  {
239  PL_drop_node(&(PH_tl_cmd_list[id]), prev, current);
240  break;
241  }
242  if (PL_get_next(current) == NULL) return PH_ACK_PACKET_NOT_FOUND;
243 
244  prev = current;
245  current = current->next;
246  }
247 
248  return PH_ACK_SUCCESS;
249 }

266  // 指定されたライン番号が存在しない場合は異常判定
267  return CCP_make_cmd_ret_without_err_code(CCP_EXEC_ILLEGAL_PARAMETER);
268  }

271  {
272  // 指定されたブロック番号が存在しない場合は異常判定
273  return CCP_make_cmd_ret_without_err_code(CCP_EXEC_ILLEGAL_PARAMETER);

277

281  (uint32_t)ack,
282  EL_ERROR_LEVEL_LOW,
283  (uint32_t)id);
284  return CCP_make_cmd_ret_without_err_code(CCP_EXEC_ILLEGAL_CONTEXT);
