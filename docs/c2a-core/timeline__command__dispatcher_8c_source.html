97 static void tlc_dispatcher_(int line_no)
98 {
99  PL_ACK ack = PL_check_tl_cmd(&(PH_tl_cmd_list[line_no]),
100  (size_t)(TMGR_get_master_total_cycle()));
101 
102  switch (ack)
103  {
104  case PL_TLC_ON_TIME:
105  // 実行予定時刻ちょうど
106  CDIS_dispatch_command(&timeline_command_dispatcher_[line_no]);
107  break;
108 
109  case PL_TLC_PAST_TIME:
110 
111  // 実行予定時刻超過
112  if (timeline_command_dispatcher_[line_no].lockout == 0)
113  {
114  // Lockout無効の場合はアノマリーを記録。
115  // Lockout有効の場合はアノマリーの連続生成を防ぐため時刻超過状態でもアノマリーを記録しない。
116 #ifndef AL_DISALBE_AT_C2A_CORE
117  AL_add_anomaly(AL_CORE_GROUP_TLCD, (uint32_t)line_no);
118 #endif
119  EL_record_event((EL_GROUP)EL_CORE_GROUP_TLCD_PAST_TIME,
120  (uint32_t)line_no,
121  EL_ERROR_LEVEL_LOW,
122  (uint32_t)CCP_get_ti(&PL_get_head(&PH_tl_cmd_list[line_no])->packet));
123 
124  if (timeline_command_dispatcher_[line_no].stop_on_error == 1)
125  {
126  // Lockout無効かつSOE有効の場合
127  // Lockout有効とし、タイムラインコマンド処理を停止。
128  timeline_command_dispatcher_[line_no].lockout = 1;
129  }
130  }
131 
132  // コマンド実行処理呼び出し。
133  // Lockoutとなった場合は呼び出してもコマンドは実行されない。
134  CDIS_dispatch_command(&timeline_command_dispatcher_[line_no]);
135  break;
136 
137  case PL_TLC_NOT_YET:
138  // 実行すべきコマンドなし。
139  break;
140 
141  default:
142  break;
143  }
144 }

217 static PH_ACK drop_tl_cmd_at_(int line_no, cycle_t time)
218 {
219  int i;
220 
221  PL_Node* prev = NULL;
222  PL_Node* current = (PL_Node*)PL_get_head(&(PH_tl_cmd_list[line_no])); // const_cast
223  int active_nodes_num = PL_count_active_nodes(&PH_tl_cmd_list[line_no]);
224 
225  if (current == NULL) return PH_TLC_NOT_FOUND;
226 
227  for (i = 0; i < active_nodes_num; ++i)
228  {
229  if (CCP_get_ti(&(current->packet)) == time)
230  {
231  PL_drop_node(&(PH_tl_cmd_list[line_no]), prev, current);
232  break;
233  }
234  if (PL_get_next(current) == NULL) return PH_TLC_NOT_FOUND;
235 
236  prev = current;
237  current = current->next;
238  }
239 
240  return PH_SUCCESS;
241 }

334  line_no = param[0];
335 
336  if (line_no >= TL_ID_MAX)

338  // 指定されたライン番号が存在しない場合は異常判定
339  return CCP_EXEC_ILLEGAL_PARAMETER;
340  }
341 
342  endian_memcpy(&block_no, &param[1], SIZE_OF_BCT_ID_T);
343 
344  if (block_no >= BCT_MAX_BLOCKS)
345  {
346  // 指定されたブロック番号が存在しない場合は異常判定
347  return CCP_EXEC_ILLEGAL_PARAMETER;
348  }
349 
350  ack = PL_deploy_block_cmd(&(PH_tl_cmd_list[line_no]), block_no, (size_t)(TMGR_get_master_total_cycle()));
351 
352  if (ack != PL_SUCCESS)
353  {
354  EL_record_event((EL_GROUP)EL_CORE_GROUP_TLCD_DEPLOY_BLOCK,
355  (uint32_t)ack,
356  EL_ERROR_LEVEL_LOW,
357  (uint32_t)line_no);
358  return CCP_EXEC_ILLEGAL_CONTEXT;
359  }
360 
361  return CCP_EXEC_SUCCESS;
