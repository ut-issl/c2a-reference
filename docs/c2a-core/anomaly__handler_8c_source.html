84 static void AH_exec_(void)
85 {
86  const AL_AnomalyRecord* ar = AL_get_record(anomaly_handler_.al_pointer.pos); // 初期値は走査開始位置
87 
88  // アノマリ個数が変化していない場合
89  if (anomaly_handler_.al_pointer.count == anomaly_logger->counter) return;
90  // 新たなアノマリが発生した場合
91  else if (anomaly_handler_.al_pointer.count < anomaly_logger->counter)
92  {
93  size_t record_id;
94 
95  for (record_id = anomaly_handler_.al_pointer.pos; record_id < anomaly_logger->header; ++record_id)
96  {
97  size_t prev_id = AH_check_rules_(record_id); // 位置合わせのため．Reprog時の対症療法なので，修正したい
98  (void)prev_id;
99  ar = AL_get_record(record_id);
100 
101  // 最初の一つ目：前回最後に記録された部分．圧縮時はここに記録される
102  if (record_id == anomaly_handler_.al_pointer.pos)
103  {
104  // もしrun_lengthが増えていれば→新たにこのアノマリが発生していれば
105  if (ar->run_length > anomaly_handler_.latest_run_length)
106  {
107  // 対応をする
108  size_t id = AH_check_rules_(record_id);
109 
110  if (id != AH_MAX_RULES)
111  {
112  AH_respond_to_anomaly_(id);
113  }
114  }
115  }
116  else
117  {
118  // anomaly_logger->headerが1以上増えている→新たな種類のアノマリが発生している
119  size_t id = AH_check_rules_(record_id);
120 
121  if (id != AH_MAX_RULES)
122  {
123  AH_respond_to_anomaly_(id);
124  }
125  }
126  }
127 
128  // 処理終了後、最新位置・回数をAnomalyLogger側と同期
129  anomaly_handler_.al_pointer.pos = anomaly_logger->header - 1;
130  anomaly_handler_.al_pointer.count = anomaly_logger->counter;
131  anomaly_handler_.latest_run_length = ar->run_length;
132  }
133  // 異常: AnomalyHandlerとAnomalyLoggerの関係が崩れている。
134  else
135  {
136  // 仕方がないので、記録をリセットし強制同期する。
137  AH_clear_log_();
138  }
139 }

142 static size_t AH_check_rules_(size_t record_id)
143 {
144  const AL_AnomalyRecord* ar = AL_get_record(record_id);
145  int is_latest = (record_id == anomaly_logger->header);
146  size_t id;
147 
148  // 全ルールを順次走査
149  for (id = 0; id < AH_MAX_RULES; ++id)
150  {
151  AH_Element ahe = anomaly_handler_.elements[id];
152 
153  // ルールが無効の場合はスキップ
154  if (!(ahe.is_active)) continue;
155  // コード不一致の場合はスキップ
156  else if (!AH_is_equal_code_(&(ahe.rule.code), &(ar->code))) continue;
157 
158  // コードに対応するルールが存在。判定条件を評価。
159  switch (ahe.rule.cond)
160  {
161  case AH_SINGLE:
162  // コードが存在した時点で無条件合致。
163  return id;
164 
165  case AH_CONTINUOUS:
166 
167  // 連続回数を超過した場合に条件合致。
168  if (ahe.rule.threshold < ar->run_length) return id;
169 
170  break;
171 
172  case AH_CUMULATE:
173 
174  // 積算回数を超過した場合に条件合致。
175  if (ahe.counter <= ar->run_length)
176  {
177  return id;
178  }
179  // 最新レコードに対する比較でない場合は積算数を更新する。
180  else if (!is_latest)
181  {
182  ahe.counter -= ar->run_length;
183  }
184 
185  break;
186  }
187  }
188 
189  return AH_MAX_RULES; // 該当なしの場合は登録上限を返す。
190 }

200 static void AH_respond_to_anomaly_(size_t id)
201 {
202  // これが呼ばれてるには，anomaly_handler_.elements[].is_active == 1は保証されている．
203

209 
210  // 実行したルールを記録し回数を更新

218  AH_respond_log_.log[AH_respond_log_.pointer].rule_num = id;
219  AH_respond_log_.pointer += 1;
220  }
221 
222  // 合致したルールは無効化する。
223  anomaly_handler_.elements[id].is_active = 0;
224 }

257 CCP_EXEC_STS Cmd_AH_REGISTER_RULE(const CommonCmdPacket* packet)
258 {
259  enum
260  {
261  ID, GROUP, LOCAL, COND, THRESHOLD, BC
262  };
263  const uint8_t* param = CCP_get_param_head(packet);
264  AH_Rule ahr;
265 
266  if (CCP_get_param_len(packet) != (5 + SIZE_OF_BCT_ID_T))
267  {
268  // パラメータ長確認(6Bytes)
269  return CCP_EXEC_ILLEGAL_LENGTH;
270  }
271  else if (param[ID] >= AH_MAX_RULES)
272  {
273  // 登録指定位置が許容範囲外
274  return CCP_EXEC_ILLEGAL_PARAMETER;
275  }
276  else if (param[COND] > AH_CUMULATE)
277  {
278  // 判定条件が定義されたものと一致しない
279  return CCP_EXEC_ILLEGAL_PARAMETER;
280  }
281 
282  ahr.code.group = (uint32_t)param[GROUP];
283  ahr.code.local = (uint32_t)param[LOCAL];
284  ahr.cond = (AH_CONDITION)param[COND];
285  ahr.threshold = (size_t)param[THRESHOLD];
286  endian_memcpy(&(ahr.bc_id), &(param[BC]), SIZE_OF_BCT_ID_T);
287 
288  AH_add_rule_((size_t)param[ID], &ahr);
289 
290  return CCP_EXEC_SUCCESS;
291 }
