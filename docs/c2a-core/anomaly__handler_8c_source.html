83 static void AH_exec_(void)
84 {
85  const AL_AnomalyRecord* ar = AL_get_record(anomaly_handler_.al_pointer.pos); // 初期値は走査開始位置
86 
87  // アノマリ個数が変化していない場合
88  if (anomaly_handler_.al_pointer.count == anomaly_logger->counter) return;
89  // 新たなアノマリが発生した場合
90  else if (anomaly_handler_.al_pointer.count < anomaly_logger->counter)
91  {
92  size_t record_id;
93 
94  for (record_id = anomaly_handler_.al_pointer.pos; record_id < anomaly_logger->header; ++record_id)
95  {
96  size_t prev_id = AH_check_rules_(record_id); // 位置合わせのため．Reprog時の対症療法なので，修正したい
97  (void)prev_id;
98  ar = AL_get_record(record_id);
99 
100  // 最初の一つ目：前回最後に記録された部分．圧縮時はここに記録される
101  if (record_id == anomaly_handler_.al_pointer.pos)
102  {
103  // もしrun_lengthが増えていれば→新たにこのアノマリが発生していれば
104  if (ar->run_length > anomaly_handler_.latest_run_length)
105  {
106  // 対応をする
107  size_t id = AH_check_rules_(record_id);
108 
109  if (id != AH_MAX_RULES)
110  {
111  AH_respond_to_anomaly_(id);
112  }
113  }
114  }
115  else
116  {
117  // anomaly_logger->headerが1以上増えている→新たな種類のアノマリが発生している
118  size_t id = AH_check_rules_(record_id);
119 
120  if (id != AH_MAX_RULES)
121  {
122  AH_respond_to_anomaly_(id);
123  }
124  }
125  }
126 
127  // 処理終了後、最新位置・回数をAnomalyLogger側と同期
128  anomaly_handler_.al_pointer.pos = anomaly_logger->header - 1;
129  anomaly_handler_.al_pointer.count = anomaly_logger->counter;
130  anomaly_handler_.latest_run_length = ar->run_length;
131  }
132  // 異常: AnomalyHandlerとAnomalyLoggerの関係が崩れている。
133  else
134  {
135  // 仕方がないので、記録をリセットし強制同期する。
136  AH_clear_log_();
137  }
138 }

141 static size_t AH_check_rules_(size_t record_id)
142 {
143  const AL_AnomalyRecord* ar = AL_get_record(record_id);
144  int is_latest = (record_id == anomaly_logger->header);
145  size_t id;
146 
147  // 全ルールを順次走査
148  for (id = 0; id < AH_MAX_RULES; ++id)
149  {
150  AH_Element ahe = anomaly_handler_.elements[id];
151 
152  // ルールが無効の場合はスキップ
153  if (!(ahe.is_active)) continue;
154  // コード不一致の場合はスキップ
155  else if (!AH_is_equal_code_(&(ahe.rule.code), &(ar->code))) continue;
156 
157  // コードに対応するルールが存在。判定条件を評価。
158  switch (ahe.rule.cond)
159  {
160  case AH_SINGLE:
161  // コードが存在した時点で無条件合致。
162  return id;
163 
164  case AH_CONTINUOUS:
165 
166  // 連続回数を超過した場合に条件合致。
167  if (ahe.rule.threshold < ar->run_length) return id;
168 
169  break;
170 
171  case AH_CUMULATE:
172 
173  // 積算回数を超過した場合に条件合致。
174  if (ahe.counter <= ar->run_length)
175  {
176  return id;
177  }
178  // 最新レコードに対する比較でない場合は積算数を更新する。
179  else if (!is_latest)
180  {
181  ahe.counter -= ar->run_length;
182  }
183 
184  break;
185  }
186  }
187 
188  return AH_MAX_RULES; // 該当なしの場合は登録上限を返す。
189 }

200 {
201  // これが呼ばれてるには，anomaly_handler_.elements[].is_active == 1は保証されている．

219  }

222  anomaly_handler_.elements[id].is_active = 0;

256 CCP_EXEC_STS Cmd_AH_REGISTER_RULE(const CommonCmdPacket* packet)
257 {
258  enum
259  {
260  ID, GROUP, LOCAL, COND, THRESHOLD, BC
261  };
262  const uint8_t* param = CCP_get_param_head(packet);
263  AH_Rule ahr;
264 
265  if (CCP_get_param_len(packet) != (5 + SIZE_OF_BCT_ID_T))
266  {
267  // パラメータ長確認(6Bytes)
268  return CCP_EXEC_ILLEGAL_LENGTH;
269  }
270  else if (param[ID] >= AH_MAX_RULES)
271  {
272  // 登録指定位置が許容範囲外
273  return CCP_EXEC_ILLEGAL_PARAMETER;
274  }
275  else if (param[COND] > AH_CUMULATE)
276  {
277  // 判定条件が定義されたものと一致しない
278  return CCP_EXEC_ILLEGAL_PARAMETER;
279  }
280 
281  ahr.code.group = (uint32_t)param[GROUP];
282  ahr.code.local = (uint32_t)param[LOCAL];
283  ahr.cond = (AH_CONDITION)param[COND];
284  ahr.threshold = (size_t)param[THRESHOLD];
285  endian_memcpy(&(ahr.bc_id), &(param[BC]), SIZE_OF_BCT_ID_T);
286 
287  AH_add_rule_((size_t)param[ID], &ahr);
288 
289  return CCP_EXEC_SUCCESS;
290 }
