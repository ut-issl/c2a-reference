13 {
14  DCM[0][0] = q[3] * q[3] + q[0] * q[0] - q[1] * q[1] - q[2] * q[2];
15  DCM[0][1] = 2.0f * (q[0] * q[1] + q[3] * q[2]);
16  DCM[0][2] = 2.0f * (q[0] * q[2] - q[3] * q[1]);
17  DCM[1][0] = 2.0f * (q[0] * q[1] - q[3] * q[2]);
18  DCM[1][1] = q[3] * q[3] - q[0] * q[0] + q[1] * q[1] - q[2] * q[2];
19  DCM[1][2] = 2.0f * (q[1] * q[2] + q[3] * q[0]);
20  DCM[2][0] = 2.0f * (q[0] * q[2] + q[3] * q[1]);
21  DCM[2][1] = 2.0f * (q[1] * q[2] - q[3] * q[0]);
22  DCM[2][2] = q[3] * q[3] - q[0] * q[0] - q[1] * q[1] + q[2] * q[2];
