267 PL_ACK PL_insert_tl_cmd(PacketList* pl, const CommonCmdPacket* packet, cycle_t now)
268 {
269  cycle_t head, tail;
270  cycle_t planed = CCP_get_ti(packet);
271 
272  if (pl->packet_type_ != PL_PACKET_TYPE_CCP) return PL_PACKET_TYPE_ERR;
273 
274  if (now > planed) return PL_TLC_PAST_TIME;
275  if (PL_is_full(pl)) return PL_LIST_FULL;
276  if (PL_is_empty(pl)) return PL_push_front(pl, packet);
277 
278  // 以下，他コマンドが登録されているとき
279  head = CCP_get_ti( (const CommonCmdPacket*)(PL_get_head(pl)->packet) );
280  tail = CCP_get_ti( (const CommonCmdPacket*)(PL_get_tail(pl)->packet) );
281 
282  if (tail < planed) // 他のどれより遅い
283  {
284  return PL_push_back(pl, packet);
285  }
286  else if (head > planed) // 他のどれより早い
287  {
288  return PL_push_front(pl, packet);
289  }
290  else if (head == planed || tail == planed) // 時刻指定が等しい
291  {
292  return PL_TLC_ALREADY_EXISTS;
293  }
294  else // 他コマンドの間に挿入操作が必要（最低でもすでに 2 個登録済みなはず）
295  {
296  uint16_t i;
297  PL_Node* prev = (PL_Node*)PL_get_head(pl); // const_cast
298  PL_Node* curr = prev->next;
299 
300  // 挿入場所探索
301  for (i = 1; i < pl->active_nodes_; ++i)
302  {
303  cycle_t curr_ti = CCP_get_ti( (const CommonCmdPacket*)(curr->packet) );
304 
305  if (curr_ti < planed)
306  {
307  prev = curr;
308  curr = prev->next;
309  }
310  else if (curr_ti > planed) // 挿入場所発見
311  {
312  PL_insert_after(pl, prev, packet);
313  return PL_SUCCESS;
314  }
315  else // 既登録コマンドと時刻指定が等しい
316  {
317  return PL_TLC_ALREADY_EXISTS;
318  }
319  }
320  }
321 
322  // NOT REACHED
323  return PL_NO_SUCH_NODE;
324 }

328 PL_ACK PL_deploy_block_cmd(PacketList* pl, const bct_id_t block, cycle_t start_at)
329 {
330  uint8_t i;
331  uint16_t j;
332  int is_cleared = 0; // リスト強制クリアの記録用変数
333  uint32_t adj = 0; // 時刻調整の累積量保存用変数
334  uint8_t bc_length;
335 
336  if (pl->packet_type_ != PL_PACKET_TYPE_CCP) return PL_PACKET_TYPE_ERR;
337 
338  if (block >= BCT_MAX_BLOCKS) return PL_BC_INACTIVE_BLOCK;
339  if (!BCE_is_active(block)) return PL_BC_INACTIVE_BLOCK;
340 
341  bc_length = BCT_get_bc_length(block);
342 
343  // リストにブロック全体を登録する余裕がない場合
344  if (PL_count_inactive_nodes(pl) < bc_length)
345  {
346  // リストをクリアし強制的に空き領域を確保する
347  PL_clear_list(pl);
348  is_cleared = 1;
349  }
350 
351  for (i = 0; i < bc_length; ++i)
352  {
353  static CommonCmdPacket temp_; // サイズが大きいため静的領域に確保
354  BCT_Pos pos;
355  PL_ACK ack = PL_SUCCESS;
356 
357  // コマンドを読みだし、TLCとして実行時刻を設定
358  BCT_make_pos(&pos, block, i);
359  BCT_load_cmd(&pos, &temp_);
360  CCP_set_ti(&temp_, (cycle_t)(start_at + adj + CCP_get_ti(&temp_)));
361  CCP_set_exec_type(&temp_, CCP_EXEC_TYPE_TL0); // BLC -> TLC // FIXME: TaskListもTL012もすべて CCP_EXEC_TYPE_TL0 になってしまうが，わかりにくくない？
362 
363  for (j = 0; j <= pl->active_nodes_; ++j)
364  {
365  // コマンドをTLCに登録を試みる
366  ack = PL_insert_tl_cmd(pl, &temp_, start_at);
367  if (ack != PL_TLC_ALREADY_EXISTS) break; // PL_SUCCESS なはず． TODO: 一応 event 発行しておく？
368 
369  // 同一時刻で既に登録されていた場合は時刻をずらして再登録
370  CCP_set_ti(&temp_, CCP_get_ti(&temp_) + 1);
371  ++adj; // 累積調整時間を更新する
372  }
373  if (ack != PL_SUCCESS) return ack;
374  }
375 
376  // リストの強制クリアを実施した場合
377  if (is_cleared == 1) return PL_BC_LIST_CLEARED;
378  // 時刻調整を行った場合
379  if (adj != 0) return PL_BC_TIME_ADJUSTED;
380 
381  return PL_SUCCESS;
382 }
