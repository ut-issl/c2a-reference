176 PL_ACK PL_insert_tl_cmd(PacketList* pli, const CTCP* packet, uint32_t now)
177 {
178  cycle_t head, tail;
179  cycle_t planed = CCP_get_ti(packet);
180 
181  if (now > planed) return PL_TLC_PAST_TIME; // 指定実行時間が既に過ぎている
182  if (PL_is_full(pli)) return PL_LIST_FULL; // 登録余裕がない
183 
184  // 何も登録されていない
185  if (PL_is_empty(pli)) return PL_push_front(pli, packet);
186 
187  // 他コマンドが登録されている
188  head = CCP_get_ti(&(PL_get_head(pli)->packet));
189  tail = CCP_get_ti(&(PL_get_tail(pli)->packet));
190 
191  if (tail < planed) // 他のどれより遅い
192  {
193  return PL_push_back(pli, packet);
194  }
195  else if (head > planed) // 他のどれより早い
196  {
197  return PL_push_front(pli, packet);
198  }
199  else if (head == planed || tail == planed) // 時刻指定が等しい
200  {
201  return PL_TLC_ALREADY_EXISTS;
202  }
203  else // 他コマンドの間に挿入操作が必要
204  {
205  int i;
206  PL_Node* prev = (PL_Node*)PL_get_head(pli); // const_cast
207  PL_Node* curr = prev->next;
208 
209  // 挿入場所探索
210  for (i = 1; i < pli->active_nodes_; ++i)
211  {
212  cycle_t curr_ti = CCP_get_ti(&(curr->packet));
213 
214  if (curr_ti < planed)
215  {
216  prev = curr;
217  curr = prev->next;
218  }
219  else if (curr_ti > planed) // 挿入場所発見
220  {
221  PL_insert_after(pli, prev, packet);
222  break;
223  }
224  else // 既登録コマンドと時刻指定が等しい
225  {
226  return PL_TLC_ALREADY_EXISTS;
227  }
228  }
229 
230  if (i == pli->active_nodes_)
231  {
232  // NOT REACHED
233  return PL_NO_SUCH_NODE;
234  }
235  }
236 
237  return PL_SUCCESS;
238 }

240 PL_ACK PL_deploy_block_cmd(PacketList* pli, const bct_id_t block, uint32_t start_at)
241 {
242  int i, j;
243  int is_cleared = 0; // リスト強制クリアの記録用変数
244  uint32_t adj = 0; // 時刻調整の累積量保存用変数
245  uint8_t length;
246 
247  if (block >= BCT_MAX_BLOCKS) return PL_BC_INACTIVE_BLOCK;
248  length = BCT_get_bc_length(block);
249 
250  if (!BCE_is_active(block)) return PL_BC_INACTIVE_BLOCK;
251 
252  // リストにブロック全体を登録する余裕がない場合
253  if (PL_count_inactive_nodes(pli) < length)
254  {
255  // リストをクリアし強制的に空き領域を確保する。
256  PL_clear_list(pli);
257  is_cleared = 1;
258  }
259 
260  for (i = 0; i < length; ++i)
261  {
262  static CTCP temp_; // サイズが大きいため静的領域に確保
263  BCT_Pos pos;
264  PL_ACK ack = PL_SUCCESS;
265 
266  // コマンドを読みだし、TLCとして実行時刻を設定。
267  pos.block = block;
268  pos.cmd = (uint8_t)i;
269  BCT_load_cmd(&pos, &temp_);
270  CCP_set_ti(&temp_, (cycle_t)(start_at + adj + CCP_get_ti(&temp_)));
271  CCP_set_exec_type(&temp_, CCP_EXEC_TYPE_TL0); // BLC -> TLC // FIXME: TaskListもTL012もすべて CCP_EXEC_TYPE_TL0 になってしまうが，わかりにくくない？
272 
273  for (j = 0; j <= pli->active_nodes_; ++j)
274  {
275  // コマンドをTLCに登録を試みる
276  ack = PL_insert_tl_cmd(pli, &temp_, start_at);
277  if (ack != PL_TLC_ALREADY_EXISTS) break;
278 
279  // 同一時刻で既に登録されていた場合は時刻をずらして再登録
280  CCP_set_ti(&temp_, CCP_get_ti(&temp_) + 1);
281  ++adj; // 累積調整時間を更新する
282  }
283  if (ack != PL_SUCCESS) return ack;
284  }
285 
286  // リストの強制クリアを実施した場合。
287  if (is_cleared == 1) return PL_BC_LIST_CLEARED;
288  // 時刻調整を行った場合。
289  if (adj != 0) return PL_BC_TIME_ADJUSTED;
290 
291  return PL_SUCCESS;
292 }
