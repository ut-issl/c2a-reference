68 CA_ACK CA_ckeck_cmd_param_len(CMD_CODE cmd_code, uint16_t param_len)
69 {
70  switch (command_analyze->cmd_table[cmd_code].param_len_type)
71  {
72  case CA_CMD_PARAM_LEN_TYPE_FIXED:
73  if (param_len != command_analyze->cmd_table[cmd_code].param_len) return CA_ACK_ERR;
74  break;
75  case CA_CMD_PARAM_LEN_TYPE_LOWER_LIMIT:
76  if (param_len < command_analyze->cmd_table[cmd_code].param_len || param_len > CCP_get_max_param_len())
77  {
78  return CA_ACK_ERR;
79  }
80  break;
81  default:
82  // CA_CMD_PARAM_LEN_TYPE_UNKNOWN
83  // なにもチェックしない
84  break;
85  }
86 
87  return CA_ACK_OK;

110  command_analyze_.cmd_table[index].param_len_type = param_len_type;
