159 CCP_EXEC_STS Cmd_CA_REGISTER_CMD(const CTCP* packet)
160 {
161  uint8_t param_size_infos[(CA_MAX_CMD_PARAM_NUM + 1) / 2];
162  CMD_CODE cmd_code = (CMD_CODE)CCP_get_param_from_packet(packet, 0, uint16_t);
163  uint32_t cmd_func = CCP_get_param_from_packet(packet, 1, uint32_t);
164  uint8_t ret;
165  uint8_t i;
166 
167  // raw パラメタなので，引数長チェック
168  if (CCP_get_param_len(packet) != 6 + sizeof(param_size_infos)) return CCP_EXEC_ILLEGAL_LENGTH;
169 
170  ret = CCP_get_raw_param_from_packet(packet, param_size_infos, sizeof(param_size_infos));
171  if (ret != sizeof(param_size_infos)) return CCP_EXEC_ILLEGAL_LENGTH;
172 
173  if (cmd_code >= CA_MAX_CMDS)
174  {
175  // 登録指定位置がコマンド数上限を超えている場合は異常判定
176  return CCP_EXEC_ILLEGAL_PARAMETER;
177  }
178 
179  // ローレベルコマンドなので，アサーションしない
180  command_analyze_.cmd_table[cmd_code].cmd_func = (CCP_EXEC_STS (*)(const CTCP*))cmd_func;
181  for (i = 0; i < sizeof(param_size_infos); ++i)
182  {
183  command_analyze_.cmd_table[cmd_code].param_size_infos[i].packed_info.bit.first = ( param_size_infos[i] & 0xf0 ) >> 4;
184  command_analyze_.cmd_table[cmd_code].param_size_infos[i].packed_info.bit.second = param_size_infos[i] & 0x0f;
185  }
186 
187  return CCP_EXEC_SUCCESS;
188 }
