170 {

190 CCP_EXEC_STS Cmd_CA_REGISTER_CMD(const CommonCmdPacket* packet)
191 {
192  uint8_t param_size_infos[(CA_MAX_CMD_PARAM_NUM + 1) / 2];
193  CMD_CODE cmd_code = (CMD_CODE)CCP_get_param_from_packet(packet, 0, uint16_t);
194  uint32_t cmd_func = CCP_get_param_from_packet(packet, 1, uint32_t);
195  uint16_t ret;
196  uint8_t i;
197 
198  // raw パラメタなので，引数長チェック
199  if (CCP_get_param_len(packet) != 6 + sizeof(param_size_infos)) return CCP_EXEC_ILLEGAL_LENGTH;
200 
201  ret = CCP_get_raw_param_from_packet(packet, param_size_infos, sizeof(param_size_infos));
202  if (ret != sizeof(param_size_infos)) return CCP_EXEC_ILLEGAL_LENGTH;
203 
204  if (cmd_code >= CA_MAX_CMDS)
205  {
206  // 登録指定位置がコマンド数上限を超えている場合は異常判定
207  return CCP_EXEC_ILLEGAL_PARAMETER;
208  }
209 
210  // ローレベルコマンドなので，アサーションしない
211  command_analyze_.cmd_table[cmd_code].cmd_func = (CCP_EXEC_STS (*)(const CommonCmdPacket*))cmd_func;
212  for (i = 0; i < sizeof(param_size_infos); ++i)
213  {
214  command_analyze_.cmd_table[cmd_code].param_size_infos[i].packed_info.bit.first = ( param_size_infos[i] & 0xf0 ) >> 4;
215  command_analyze_.cmd_table[cmd_code].param_size_infos[i].packed_info.bit.second = param_size_infos[i] & 0x0f;
216  }
217 
218  return CCP_EXEC_SUCCESS;
219 }
