117 ObcTime TMGR_get_ObcTime_from_unixtime(const double unixtime)
118 {
119  double diff_double = unixtime - OBCT_unixtime_info_.unixtime_at_ti0;
120  ObcTime res;
121  uint32_t diff;
122  cycle_t cycle_diff;
123  step_t step_diff;
124 
125  if (diff_double < 0) // あり得ない, おかしい
126  {
127  return res = OBCT_create(0, 0, 0);
128  }
129 
130  diff = (uint32_t)(diff_double * 1000.0 + 1e-4); // msオーダーだがそんなに大きくないことを想定, 1e-4は数値誤差対策（.999がj切り捨てられるのを防ぐ）
131  cycle_diff = diff / (OBCT_STEP_IN_MSEC * OBCT_STEPS_PER_CYCLE);
132  step_diff = (diff - cycle_diff * (OBCT_STEP_IN_MSEC * OBCT_STEPS_PER_CYCLE)) / OBCT_STEP_IN_MSEC;
133 
134  res.total_cycle = cycle_diff;
135  res.mode_cycle = 0; // 取得出来ないので0とする
136  res.step = step_diff;
137 
138  return res;
139 }
