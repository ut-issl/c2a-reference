113 ObcTime TMGR_get_obc_time_from_unixtime(const double unixtime)
114 {
115  double diff = unixtime - TMGR_get_obct_unixtime_info().unixtime_at_ti0; // 秒単位
116  cycle_t cycle_diff;
117  step_t step_diff;
118  ObcTime res;
119 
120  if (diff < 0) // あり得ない, おかしい
121  {
122  return OBCT_create(0, 0, 0);
123  }
124 
125  cycle_diff = (cycle_t)(diff * OBCT_CYCLES_PER_SEC); // cycle未満は切り捨て
126  step_diff = (step_t)((diff * OBCT_CYCLES_PER_SEC - cycle_diff) * OBCT_STEPS_PER_CYCLE); // step未満は切り捨て
127 
128  res.total_cycle = cycle_diff;
129  res.mode_cycle = 0; // 取得出来ないので0とする
130  res.step = step_diff;
131 
132  return res;
133 }
