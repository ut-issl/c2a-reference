95 ObcTime TMGR_get_ObcTime_from_unix_time(const double unix_time)
96 {
97  double diff_double = unix_time - OBCT_unix_time_info->unix_time_on_ti0;
98  ObcTime res;
99  uint32_t diff;
100  cycle_t cycle_diff;
101  step_t step_diff;
102 
103  if (diff_double < 0) // あり得ない, おかしい
104  {
105  return res = OBCT_create(0, 0, 0);
106  }
107 
108  diff = (uint32_t)(diff_double * 1000.0 + 1e-4); // msオーダーだがそんなに大きくないことを想定, 1e-4は数値誤差対策（.999がj切り捨てられるのを防ぐ）
109  cycle_diff = diff / (OBCT_STEP_IN_MSEC * OBCT_STEPS_PER_CYCLE);
110  step_diff = (diff - cycle_diff * (OBCT_STEP_IN_MSEC * OBCT_STEPS_PER_CYCLE)) / OBCT_STEP_IN_MSEC;
111 
112  res.total_cycle = cycle_diff;
113  res.mode_cycle = 0; // 取得出来ないので0とする
114  res.step = step_diff;
115 
116  return res;
117 }
