193 ObcTime TMGR_get_obc_time_from_unixtime(const double unixtime)
194 {
195  double diff_in_cycle = TMGR_get_precise_ti_from_unixtime(unixtime);
196  cycle_t cycle;
197  step_t step;
198  ObcTime res;
199 
200  cycle = (cycle_t)diff_in_cycle; // cycle未満は切り捨て
201  step = (step_t)((diff_in_cycle - cycle) * OBCT_STEPS_PER_CYCLE); // step未満は切り捨て
202 
203  res.total_cycle = cycle;
204  res.mode_cycle = 0; // 取得出来ないので0とする

256 CCP_CmdRet Cmd_TMGR_UPDATE_UNIXTIME(const CommonCmdPacket* packet)
257 {
258  ObcTime time;
259  double unixtime = CCP_get_param_from_packet(packet, 0, double);
260  cycle_t total_cycle = CCP_get_param_from_packet(packet, 1, cycle_t);
261  step_t step = CCP_get_param_from_packet(packet, 2, cycle_t);
262  TMGR_ACK ack;
263 
264  if (unixtime < 0) return CCP_make_cmd_ret_without_err_code(CCP_EXEC_ILLEGAL_PARAMETER);
265  if (total_cycle >= OBCT_MAX_CYCLE) return CCP_make_cmd_ret_without_err_code(CCP_EXEC_ILLEGAL_PARAMETER);
266  if (step >= OBCT_STEPS_PER_CYCLE) return CCP_make_cmd_ret_without_err_code(CCP_EXEC_ILLEGAL_PARAMETER);
267 
268  time.total_cycle = total_cycle;
269  time.step = step;
270  time.mode_cycle = 0; // 必要ないので0とする
271 
272  ack = TMGR_update_unixtime(unixtime, &time);
273 
274  return TMGR_conv_tmgr_ack_to_ccp_cmd_ret_(ack);
