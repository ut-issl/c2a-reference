177 ObcTime TMGR_get_obc_time_from_unixtime(const double unixtime)
178 {
179  double diff_in_cycle = (unixtime - TMGR_get_unixtime_at_ti0()) * TMGR_get_precice_cycles_per_sec();
180  cycle_t cycle;
181  step_t step;
182  ObcTime res;
183 
184  if (diff_in_cycle < 0) // あり得ない, おかしい
185  {
186  return OBCT_create(0, 0, 0);
187  }
188  cycle = (cycle_t)diff_in_cycle; // cycle未満は切り捨て
189  step = (step_t)((diff_in_cycle - cycle) * OBCT_STEPS_PER_CYCLE); // step未満は切り捨て
190 
191  res.total_cycle = cycle;
192  res.mode_cycle = 0; // 取得出来ないので0とする
193  res.step = step;
194 
195  return res;
196 }

263 
264  time.total_cycle = total_cycle;
265  time.step = step;
266  time.mode_cycle = 0; // 必要ないので0とする
267
