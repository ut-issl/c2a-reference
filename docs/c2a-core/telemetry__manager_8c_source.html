451 static CCP_CmdRet TLM_MGR_clear_cmds_based_on_role_(CMD_CODE cmd_code,
452  TLM_MGR_BC_ROLE bc_role,
453  TLM_MGR_RegisterInfo* register_info)
454 {
455  uint16_t exec_counter;
456  uint8_t cmd_table_idx;
457  uint8_t bc_num = register_info->cmd_table_idxes_size;
458 
459  switch (DCU_check_in(cmd_code, &exec_counter))
460  {
461  case DCU_STATUS_FINISHED: // FALLTHROUGH
462  case DCU_STATUS_PROGRESS:
463  break;
464  default:
465  // DCU_STATUS_ABORTED_BY_ERR
466  // DCU_STATUS_ABORTED_BY_CMD
467  // がここに
468  return CCP_make_cmd_ret_without_err_code(CCP_EXEC_ILLEGAL_CONTEXT);
469  }
470 
471  if (exec_counter == bc_num)
472  {
473  register_info->registered_cmd_num = 0;
474  DCU_report_finish(cmd_code, CCP_EXEC_SUCCESS);
475  return CCP_make_cmd_ret_without_err_code(CCP_EXEC_SUCCESS);
476  }
477 
478  cmd_table_idx = register_info->cmd_table_idxes[exec_counter];
479  TLM_MGR_clear_bc_to_nop_(telemetry_manager_.cmd_table.cmd_blocks[cmd_table_idx].bc_id);
480  memset(telemetry_manager_.cmd_table.cmd_blocks[cmd_table_idx].cmds,
481  0x00,
482  sizeof(TLM_MGR_CmdElem) * TLM_MGR_MAX_CMD_NUM_PER_BC);
483 
484  // 再帰実行
485  if (DCU_register_next(cmd_code, NULL, 0) != DCU_ACK_OK)
486  {
487  DCU_report_err(cmd_code, CCP_EXEC_ILLEGAL_CONTEXT);
488  return CCP_make_cmd_ret_without_err_code(CCP_EXEC_ILLEGAL_CONTEXT);
489  }
490 
491  return CCP_make_cmd_ret_without_err_code(CCP_EXEC_SUCCESS);
492 }
