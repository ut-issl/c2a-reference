242 static TLM_MGR_ERR_CODE TLM_MGR_calc_register_info_from_bc_info_(void)
243 {
244  uint8_t bc_info_idx;
245  TLM_MGR_RegisterInfo* register_info_master;
246 
247  TLM_MGR_clear_register_info_all_(); // TODO: 高速化のために消してもいいかも？
248 
249  for (bc_info_idx = 0; bc_info_idx < TLM_MGR_USE_BC_NUM; ++bc_info_idx)
250  {
251  TLM_MGR_ERR_CODE ret = TLM_MGR_ERR_CODE_OK;
252  switch (telemetry_manager_.bc_info[bc_info_idx].bc_type)
253  {
254  case TLM_MGR_BC_TYPE_MASTER:
255  ret = TLM_MGR_add_bc_info_to_register_info_(&telemetry_manager_.register_info.master, bc_info_idx);
256  break;
257  case TLM_MGR_BC_TYPE_HK_TLM:
258  ret = TLM_MGR_add_bc_info_to_register_info_(&telemetry_manager_.register_info.hk_tlm, bc_info_idx);
259  break;
260  case TLM_MGR_BC_TYPE_SYSTEM_TLM:
261  ret = TLM_MGR_add_bc_info_to_register_info_(&telemetry_manager_.register_info.system_tlm, bc_info_idx);
262  break;
263  case TLM_MGR_BC_TYPE_HIGH_FREQ_TLM:
264  ret = TLM_MGR_add_bc_info_to_register_info_(&telemetry_manager_.register_info.high_freq_tlm, bc_info_idx);
265  break;
266  case TLM_MGR_BC_TYPE_LOW_FREQ_TLM:
267  ret = TLM_MGR_add_bc_info_to_register_info_(&telemetry_manager_.register_info.low_freq_tlm, bc_info_idx);
268  break;
269  case TLM_MGR_BC_TYPE_RESERVE:
270  ret = TLM_MGR_add_bc_info_to_register_info_(&telemetry_manager_.register_info.reserve, bc_info_idx);
271  break;
272  default:
273  return TLM_MGR_ERR_CODE_OTHER_ERR;
274  break;
275  }
276  if (ret != TLM_MGR_ERR_CODE_OK) return ret;
277  }
278 
279  register_info_master = &telemetry_manager_.register_info.master;
280 
281  if (register_info_master->bc_info_idx_used_num == 0) return TLM_MGR_ERR_CODE_MASTER_IS_ABSENT;
282  if (register_info_master->bc_info_idx_used_num > 1) return TLM_MGR_ERR_CODE_MASTER_DUPLICATED;
283 
284  bc_info_idx = register_info_master->bc_info_idx[0];
285  telemetry_manager_.master_bc_id = telemetry_manager_.bc_info[bc_info_idx].bc_id;
286  return TLM_MGR_ERR_CODE_OK;
287 }

315 static TLM_MGR_ERR_CODE TLM_MGR_register_generate_tlm_(TLM_MGR_RegisterInfo* register_info, const uint8_t* param)
316 {
317  uint8_t bc_info_idx = register_info->bc_info_idx[register_info->tlm_register_pointer_to_idx_of_bc_info_idx];
318  bct_id_t bc_id = telemetry_manager_.bc_info[bc_info_idx].bc_id;
319  uint8_t bc_cmd_pos = register_info->tlm_register_pointer_to_bc_cmd_idx;
320  BCT_Pos bc_register_pos;
321  CCP_UTIL_ACK ccp_util_ack;
322  BCT_ACK bct_ack;
323 
324  if (bc_cmd_pos >= TLM_MGR_MAX_TLM_NUM_PER_BC) return TLM_MGR_ERR_CODE_CMD_FULL;
325 
326  BCT_make_pos(&bc_register_pos, bc_id, bc_cmd_pos);
327 
328  ccp_util_ack = CCP_form_tlc(&TLM_MGR_packet_,
329  (cycle_t)bc_cmd_pos,
330  Cmd_CODE_GENERATE_TLM,
331  param,
332  CA_get_cmd_param_min_len(Cmd_CODE_GENERATE_TLM));
333  if (ccp_util_ack != CCP_UTIL_ACK_OK) return TLM_MGR_ERR_CODE_OTHER_ERR;
334 
335  // すでに NOP で埋められており，かつ activate 済なので，直接 BCT にコマンドを挿入できる．
336  bct_ack = BCT_overwrite_cmd(&bc_register_pos, &TLM_MGR_packet_);
337  if (bct_ack != BCT_SUCCESS) return TLM_MGR_ERR_CODE_OTHER_ERR;
338 
339  register_info->tlm_register_pointer_to_idx_of_bc_info_idx++;
340  if (register_info->tlm_register_pointer_to_idx_of_bc_info_idx >= register_info->bc_info_idx_used_num)
341  {
342  register_info->tlm_register_pointer_to_idx_of_bc_info_idx %= register_info->bc_info_idx_used_num;
343  register_info->tlm_register_pointer_to_bc_cmd_idx++;
344  }
345 
346  return TLM_MGR_ERR_CODE_OK;
347 }

350 static void TLM_MGR_load_master_bc_(void)
351 {
352  cycle_t ti = 1; // 1 - 9 までの 9 個登録する． 10 はdeploy
353  uint8_t bc_info_idx;
354 
355  for (bc_info_idx = 0; bc_info_idx < TLM_MGR_USE_BC_NUM; ++bc_info_idx)
356  {
357  switch (telemetry_manager_.bc_info[bc_info_idx].bc_type)
358  {
359  case TLM_MGR_BC_TYPE_HK_TLM: // FALLTHROUGH
360  case TLM_MGR_BC_TYPE_SYSTEM_TLM: // FALLTHROUGH
361  case TLM_MGR_BC_TYPE_HIGH_FREQ_TLM: // FALLTHROUGH
362  case TLM_MGR_BC_TYPE_RESERVE:
363  BCL_tool_register_combine(ti, telemetry_manager_.bc_info[bc_info_idx].bc_id);
364  ti++;
365  break;
366  case TLM_MGR_BC_TYPE_LOW_FREQ_TLM:
367  BCL_tool_register_rotate(ti, telemetry_manager_.bc_info[bc_info_idx].bc_id);
368  ti++;
369  break;
370  default:
371  // TLM_MGR_BC_TYPE_MASTER がここに
372  break;
373  }
374  }
375 
376  BCL_tool_register_deploy(TLM_MGR_USE_BC_NUM, telemetry_manager_.master_bc_id, TLCD_ID_DEPLOY_TLM);
