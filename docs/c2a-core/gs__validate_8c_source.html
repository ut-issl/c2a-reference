158 static GS_VALIDATE_ERR GS_check_ad_cmd_(const TcTransferFrame* tctf)
159 {
160  GS_VALIDATE_ERR ack;
161  const TcSegment* tc_segment = TCTF_get_tc_segment(tctf);
162  uint8_t seq_diff = (GS_RECEIVE_WINDOW + TCTF_get_frame_seq_num(tctf) - gs_validate_info_.type_a_counter) % GS_RECEIVE_WINDOW;
163 
164  if (gs_validate_info_.lockout_flag) return GS_VALIDATE_ERR_IN_LOCKOUT;
165 
166  ack = GS_check_tc_segment_(tc_segment);
167  if (ack != GS_VALIDATE_ERR_OK) return ack;
168 
169  // N(R) == V(R)なら正常受信
170  if (seq_diff == 0)
171  {
172  gs_validate_info_.retransmit_flag = 0;
173  ++gs_validate_info_.type_a_counter;
174  return GS_VALIDATE_ERR_OK;
175  }
176  // 送信側が行き過ぎているのでパケット破棄して再送要求
177  else if (seq_diff <= gs_validate_info_.positive_window_width)
178  {
179  gs_validate_info_.retransmit_flag = 1;
180  return GS_VALIDATE_ERR_FARM1_POSITIVE_WINDOW_AREA;
181  }
182  // もう既に送られているので破棄
183  else if ((256 - gs_validate_info_.positive_window_width) <= seq_diff)
184  {
185  return GS_VALIDATE_ERR_FARM1_NEGATIVE_WINDOW_AREA;
186  }
187  // 許容 window から外れているので破棄してロックアウト, BC コマンドでロックアウト解除要求を行う
188  // ロックアウトするので一切のコマンドを受け付けない
189  else
190  {
191  gs_validate_info_.lockout_flag = 1;
192  return GS_VALIDATE_ERR_FARM1_LOCKOUT_AREA;
193  }
194 
195  // not reaced
196  return GS_VALIDATE_ERR_UNKNOWN;
197 }

213 static GS_VALIDATE_ERR GS_check_bc_cmd_(const TcTransferFrame* tctf)
214 {
215  uint16_t length = TCTF_get_frame_len(tctf);
216  uint8_t offset = TCTF_HEADER_SIZE + TCTF_FECF_SIZE;
217  const TcSegment* tc_segment = TCTF_get_tc_segment(tctf);
218 
219  // BC コマンドは COP-1 制御の制御用コマンドで特殊なため少し構造が異なる
220  // Unlock
221  if (tc_segment->packet[0] == TCTF_BC_CMD_CODE_UNLOCK && length == offset + 1)
222  {
223  gs_validate_info_.lockout_flag = 0;
224  gs_validate_info_.retransmit_flag = 0;
225 
226  ++gs_validate_info_.type_b_counter;
227  }
228  // SET V(R)
229  else if (tc_segment->packet[0] == TCTF_BC_CMD_CODE_SET_VR_1ST_BYTE &&
230  tc_segment->packet[1] == TCTF_BC_CMD_CODE_SET_VR_2ND_BYTE &&
231  length == offset + 3)
232  {
233  if (gs_validate_info_.lockout_flag == 0)
234  {
235  gs_validate_info_.type_a_counter = tc_segment->packet[2];
236  gs_validate_info_.retransmit_flag = 0;
237  }
238 
239  // Lockout状態でもType-B Counterの値は更新する
240  ++gs_validate_info_.type_b_counter;
241  }
242  else
243  {
244  return GS_VALIDATE_ERR_INVALID_BC_CMD;
245  }
246 
247  return GS_VALIDATE_ERR_OK;
248 }

261 
262  // FIXME: No RF Available Flag
263  // **** **** **** **** x*** **** **** ****
264  // FIXME: NO Bit Lock Flag
265  // **** **** **** **** *x** **** **** ****
266 
267  // Lockout flag
268  if (gs_validate_info_.lockout_flag)
269  {

271  }
272  // Retransmit
273  if (gs_validate_info_.retransmit_flag)
274  {
275  clcw |= 0x00000800; // **** **** **** **** **** 1*** **** ****
276  }
