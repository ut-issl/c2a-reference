140 static GS_VALIDATE_ERR GS_check_fecw_(const uint8_t* data, size_t len)
141 {
142  int i, j;
143  uint16_t shift_reg = 0xffff; // 初期値は全bitが1
144  uint16_t xor_tap = 0x1021; // LSBは常時0とXORをとっていると考える。
145 
146  // データ長だけループ
147  for (i = 0; i < len; ++i)
148  {
149  // MSB位置をshift_regと揃えるため8bit左シフト
150  uint16_t tmp = (uint16_t)(data[i] << 8);
151 
152  // ビット長だけループ
153  for (j = 0; j < 8; ++j)
154  {
155  // MSB同士のXORを比較
156  if ((shift_reg ^ tmp) & 0x8000)
157  {
158  // 結果が1の場合はシフト+XOR
159  shift_reg <<= 1;
160  shift_reg ^= xor_tap;
161  }
162  else
163  {
164  // 結果が0の場合はシフトのみ
165  shift_reg <<= 1;
166  }
167 
168  // 次ビット評価のためシフト
169  tmp <<= 1;
170  }
171  }
172 
173  // データ+FECWが正常なら結果は0となる
174  if (shift_reg != 0) return GS_VALIDATE_ERR_FECW_MISSMATCH;
175 
176  return GS_VALIDATE_ERR_OK;
177 }

179 static GS_VALIDATE_ERR GS_check_ad_cmd_(const TCF* tc_frame)
180 {
181  GS_VALIDATE_ERR ack;
182  int seq_diff;
183 
184  if (gs_validate_info_.lockout_flag) return GS_VALIDATE_ERR_IN_LOCKOUT;
185 
186  ack = GS_check_tcf_contents_(tc_frame);
187  if (ack != GS_VALIDATE_ERR_OK) return ack;
188 
189  seq_diff = TCF_get_frame_seq_num(tc_frame) - gs_validate_info_.type_a_counter;
190 
191  if (seq_diff < 0)
192  {
193  // Sequence Counterの値はmod-256なのでseq_diffの値が負の場合は
194  // 256を足して値の範囲を[1, 256]に変換
195  seq_diff += 256;
196  }
197 
198  if (seq_diff == 0)
199  {
200  // seq_diffが0、すなわちN(R) == V(R)なら正常受信
201  // 再送要求フラグのクリアとシーケンス数のインクリメント
202  gs_validate_info_.retransmit_flag = 0;
203  ++gs_validate_info_.type_a_counter;
204  }
205  else
206  {
207  if (seq_diff < gs_validate_info_.positive_window_width)
208  {
209  gs_validate_info_.retransmit_flag = 1;
210  return GS_VALIDATE_ERR_FARM1_POSITIVE_WINDOW_AREA;
211  }
212  else if (seq_diff < (256 - gs_validate_info_.positive_window_width))
213  {
214  gs_validate_info_.retransmit_flag = 1;
215  return GS_VALIDATE_ERR_FARM1_LOCKOUT_AREA;
216  }
217  else
218  {
219  if (seq_diff == 255) return GS_VALIDATE_ERR_FARM1_SAME_NUMBER;
220  else return GS_VALIDATE_ERR_FARM1_NEGATIVE_WINDOW_AREA;
221  }
222  }
223 
224  return GS_VALIDATE_ERR_OK;
225 }

227 static GS_VALIDATE_ERR GS_check_bc_cmd_(const TCF* tc_frame)
228 {
229  // BCコマンドの種別を判定し、処理する。
230  // TCFの構成がAD/BDコマンドに特化した形となっているため、
231  // TCSやTCPのデータ構造を読み替えて処理を行っている。
232  if (tc_frame->tcs.header[0] == TCF_BC_CMD_CODE_UNLOCK)
233  {
234  // Unlockコマンドの場合はLockoutとRetransmitフラグをクリア
235  gs_validate_info_.lockout_flag = 0;
236  gs_validate_info_.retransmit_flag = 0;
237 
238  // Type-B Coutnerの値を更新して処理終了
239  ++gs_validate_info_.type_b_counter;
240  }
241  else if ((tc_frame->tcs.header[0] == TCF_BC_CMD_CODE_SET_VR_0)
242  && (tc_frame->tcs.tcp.packet[0] == TCF_BC_CMD_CODE_SET_VR_1))
243  {
244  // SET V(R)コマンドの場合
245  if (gs_validate_info_.lockout_flag == 0)
246  {
247  // Lockout状態でない場合はType-Aカウンタの値を指定値に設定し
248  // Retransmitフラグをクリア
249  gs_validate_info_.type_a_counter = tc_frame->tcs.tcp.packet[1];
250  gs_validate_info_.retransmit_flag = 0;
251  }
252 
253  // Type-B Coutnerの値を更新して処理終了
254  // Lockout状態でもType-B Counterの値は更新する
255  ++gs_validate_info_.type_b_counter;
256  }
257  else
258  {
259  // 上記以外の場合は不正と判断し異常終了
260  return GS_VALIDATE_ERR_INVALID_BC_CMD;
261  }
262 
263  return GS_VALIDATE_ERR_OK;
264 }
